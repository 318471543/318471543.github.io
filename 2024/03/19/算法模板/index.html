数据结构

Trie

封装模板：

const int N = 1e5 + 10, bit = 31, M = bit * N;
struct Trie
{
    int idx, son[M][2];
    void clear () { // 清空
        for (int i = 0; i <= idx; i++)
            son[i][0] = son[i][1] = 0;
        idx = 0;
    }
    void insert(int x) {
        int p = 0;
        for(int i = bit;i >= 0;i--) {
            int u = x >> i & 1;
            if(!son[p][u])  son[p][u] = ++idx;
            p = son[p][u];
        }
    }
    int query(int x) {
        int p = 0,res = 0;
        for(int i = bit;i >= 0;i--) {
            int u = x >> i & 1;
            if(son[p][!u]) {
                p = son[p][!u];
                res = res * 2 + !u;
            }
            else {
                p = son[p][u];
                res = res * 2 + u;
            }
        }
        return res;
    }
}trie;

后缀自动机

封装模板：

const int N = 1e6 * 2 + 10; // 空间开2倍
struct SAM
{
    int tot,last;
    struct Node
    {
        int len,fa;
        int ch[26];
        void init(){//结点初始化
            len = fa = 0;
            memset(ch,0,sizeof(ch));
        }
    }node[N];
    SAM() {tot = 1,last = 1;}
    void init() {//清空区间[1,tot]
        for(int i = 1;i <= tot;i++) node[i].init();
        tot = last = 1;
    }
    void extend(int c) {//添加一个字符(c - 'a')
        int p = last,np = last = ++tot;
        node[np].len = node[p].len + 1;
        for(;p && !node[p].ch[c];p = node[p].fa)    node[p].ch[c] = np;
        if(!p)  node[np].fa = 1;
        else
        {
            int q = node[p].ch[c];
            if(node[q].len == node[p].len + 1)   node[np].fa = q;
            else
            {
                int nq = ++ tot;
                node[nq] = node[q],node[nq].len = node[p].len + 1;
                node[q].fa = node[np].fa = nq;
                for(;p && node[p].ch[c] == q;p = node[p].fa)    node[p].ch[c] = nq;
            }
        }
    }
}sam;

推荐题目：最长公共子串

代码：

#include<bits/stdc++.h>

using namespace std;

const int N = 1e4 * 2 + 10; // 空间开2倍

struct SAM
{
    int tot,last;
    struct Node
    {
        int len,fa;
        int ch[26];
        void init(){//结点初始化
            len = fa = 0;
            memset(ch,0,sizeof(ch));
        }
    }node[N];
    SAM() {tot = 1,last = 1;}
    void init() {//清空区间[1,tot]
        for(int i = 1;i <= tot;i++) node[i].init();
        tot = last = 1;
    }
    void extend(int c) {//添加一个字符(c - 'a')
        int p = last,np = last = ++tot;
        node[np].len = node[p].len + 1;
        for(;p && !node[p].ch[c];p = node[p].fa)    node[p].ch[c] = np;
        if(!p)  node[np].fa = 1;
        else
        {
            int q = node[p].ch[c];
            if(node[q].len == node[p].len + 1)   node[np].fa = q;
            else
            {
                int nq = ++ tot;
                node[nq] = node[q],node[nq].len = node[p].len + 1;
                node[q].fa = node[np].fa = nq;
                for(;p && node[p].ch[c] == q;p = node[p].fa)    node[p].ch[c] = nq;
            }
        }
    }
}sam;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int n;
    cin >> n;

    string s;
    cin >> s;

    for (auto c : s)    sam.extend(c - 'a');

    vector<vector<int>> g(N + 1);
    for (int i = 2; i <= sam.tot; i++)  g[sam.node[i].fa].push_back(i);

    vector<int> res(N + 10, 1e9);
    for (int i = 0; i < n - 1; i++)
    {
        cin >> s;
        vector<int> now(N + 10);
        function<void(int)> dfs = [&](int u) -> void
        {
            for (auto v : g[u]) 
            {
                dfs(v);
                now[u] = min(sam.node[u].len, max(now[u], now[v]));
            }
        };
        int p = 1, len = 0;
        for (auto c : s)
        {
            while (p > 1 && !sam.node[p].ch[c - 'a'])   p = sam.node[p].fa, len = sam.node[p].len;
            if (sam.node[p].ch[c - 'a'])    p = sam.node[p].ch[c - 'a'], len++;
            now[p] = max(now[p], len);
        }
        dfs(1);
        for (int i = 1; i <= sam.tot; i++)  res[i] = min(res[i], now[i]);
    }

    cout << *max_element(res.begin() + 1, res.begin() + sam.tot + 1) << "\n";
    return 0;
}

线段树 + 后缀自动机

推荐题目： Hacker

代码：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 1e6 + 10;
int w[N];
char str[N],s[N];

struct SegmentTree
{
    int l,r;
    ll sum,rs;
}tr[N << 2];
struct Suffix_Automation
{
    int len,fa;
    int ch[26];
}node[N << 1];
int last = 1,tot = 1;
int h[N],e[N],ne[N],idx;

void pushup(int u)
{
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
    tr[u].rs = max(tr[u << 1 | 1].rs,tr[u << 1].rs + tr[u << 1 | 1].sum);
}

void build(int u,int l,int r)
{
    tr[u] = {l,r};
    if(l == r)  tr[u].sum = tr[u].rs = w[l];
    else
    {
        int mid = l + r >> 1;
        build(u << 1,l,mid),build(u << 1 | 1,mid + 1,r);
        pushup(u);
    } 
}

SegmentTree query(int u,int l,int r)
{
    if(l <= tr[u].l && r >= tr[u].r)    return tr[u];
    
    int mid = tr[u].l + tr[u].r >> 1;
    if(l > mid) return query(u << 1 | 1,l,r);
    else if(r <= mid)    return query(u << 1,l,r);
    else
    {
        SegmentTree res;
        SegmentTree left = query(u << 1,l,r);
        SegmentTree right = query(u << 1 | 1,l,r);
        res.sum = left.sum + right.sum;
        res.rs = max(right.rs,left.rs + right.sum);
        return res;
    }   
}

ll query(int l,int r)
{
    return query(1,l,r).rs;
}

void extend(int c)
{
    int p = last,np = last = ++tot;
    node[np].len = node[p].len + 1;
    for(;p && !node[p].ch[c];p = node[p].fa)  node[p].ch[c] = np;
    if(!p)  node[np].fa = 1;
    else
    {
        int q = node[p].ch[c];
        if(node[q].len == node[p].len + 1)  node[np].fa = q;
        else
        {
            int nq = ++ tot;
            node[nq] = node[q],node[nq].len = node[p].len + 1;
            node[q].fa = node[np].fa = nq;
            for(;p && node[p].ch[c] == q;p = node[p].fa)    node[p].ch[c] = nq;
        }
    }
}

void insert(char *str)
{
    for(int i = 0;str[i];i++)   extend(str[i] - 'a');
}

ll solve(char *s)
{
    int p = 1,t = 0;
    ll res = 0;
    for(int i = 0;s[i];i++)
    {
        int c = s[i] - 'a';
        while(p > 1 && !node[p].ch[c])    p = node[p].fa,t = node[p].len;
        if(node[p].ch[c])   p = node[p].ch[c],t++;
        if(t)   res = max(res,query(i + 1 - t + 1,i + 1));
    }
    return  res;
}

signed main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    int n,m,k;
    cin >> n >> m >> k;
    cin >> str;

    for(int i = 1;i <= m;i++)   cin >> w[i];
 
    insert(str);
    build(1,1,m);

    while (k--)
    {
        cin >> s;
        cout << solve(s) << "\n";
    }

    return 0;
}
