#include<bits/stdc++.h>

using namespace std;

using ll = long long;

constexpr int mod = 1000000007;
constexpr int N = 1e5 + 10;

ll fact[N], infact[N];

ll fpow(ll a, ll b)
{
    ll res = 1;
    while (b)
    {
        if (b & 1)  res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    fact[0] = infact[0] = 1;
    for (int i = 1; i < N; i++)
    {
        fact[i] = fact[i - 1] * i % mod;
        infact[i] = fpow(fact[i], mod - 2);
    }

    int n;
    cin >> n;

    vector<int> d(n + 1);
    for (int i = 1; i < n; i++)
    {
        int a, b;
        cin >> a >> b;
        d[a]++, d[b]++;
    }

    int L, R;
    cin >> L >> R;

    map<int, int> mpd;
    for (int i = 1; i <= n; i++)
        mpd[d[i]]++;

    auto C = [&](int a, int b) -> ll
    {
        return fact[b] * infact[a] % mod * infact[b - a] % mod;
    };

    ll res = 0;
    if (L <= 1) res += n;
    if (L <= 2 && R >= 2)
    {
        res += n - 1;
    }
    for (auto [du, cnt] : mpd)
    {
        // if (du <= 1)    continue;
        if (du >= L - 1)
        {
            int l = max(2, L - 1);
            int r = min(du, R - 1);
            ll ans = 0;
            
            for (int i = l; i <= r; i++)
                ans = (ans + C(i, du)) % mod;

            // cout << du << " " << cnt << " " << ans << " " << l << " " << r << "\n";
            res = (res + ans * cnt % mod) % mod;
        }
    }

    cout << res;
    return 0;
}
