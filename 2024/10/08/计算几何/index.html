#include<bits/stdc++.h>

using namespace std;

const double eps = 1e-8, pi = acos(-1);

template<class T>
int sign(T x) { // 符号函数
    if(fabs(x) < eps)	return 0;
    return x < 0 ? -1 : 1;
}

template<class T>
int cmp(T x,T y) { // 比较函数
    if(fabs(x - y) < eps)	return 0; // 两数相等
    return x < y ? -1 : 1;
}

template<class T>
struct Point
{
	T x, y;
	Point(T _x = 0, T _y = 0) : x(_x), y(_y) {}

    bool operator<(Point p) const {
        int c = cmp(x, p.x);
        if (c)  return c == -1;
        return cmp(y, p.y) == -1;
    }
    bool operator==(Point p) const {
        return cmp(x, p.x) == 0 && cmp(y, p.y) == 0;
    }

	Point &operator+=(Point p) & {
		x += p.x;
		y += p.y;
		return *this;
	}
	Point &operator-=(Point p) & {
		x -= p.x;
		y -= p.y;
		return *this;
	}
	Point &operator*=(T v) & {
		x *= v;
		y *= v;
		return *this;
	}
	Point &operator/=(T v) & {
		x /= v;
		y /= v;
		return *this;
	}
	Point operator-() const {
		return Point(-x, -y);
	}
	friend Point operator+(Point a, Point b) {
		return a += b;
	}
	friend Point operator-(Point a, Point b) {
		return a -= b;
	}
	friend Point operator*(Point a, T b) {
		return a *= b;
	}
	friend Point operator/(Point a, T b) {
		return a /= b;
	}
	friend Point operator*(T a, Point b) {
		return b *= a;
	}
	friend istream &operator>>(istream &is, Point &p) {
		return is >> p.x >> p.y;
	}
	friend ostream &operator<<(ostream &os, Point p) {
		return os << "(" << p.x << ", " << p.y << ")";
	}
};
using P = Point<double>;
using Vector = Point<double>;

template<class T>
struct Line {
    Point<T> a, b;
    Line(Point<T> _a = Point<T>(), Point<T> _b = Point<T>()) : a(_a), b(_b) {}
};
using L = Line<double>;

template<class T>
T dot(Point<T> a, Point<T> b) { // 点积
	return a.x * b.x + a.y * b.y;
}

template<class T>
T cross(Point<T> a, Point<T> b) { // 叉积
	return a.x * b.y - a.y * b.x;
}

template<class T>
T length(Point<T> p) { // 模长
	return sqrt(dot(p, p));
}

template<class T>
T area(Point<T> a, Point<T> b, Point<T> c) { // 计算两个向量构成的平行四边形有向面积
    return cross(b - a,c - a);
}

template<class T>
int crossSign(Point<T> a, Point<T> b, Point<T> c) {
    return sign(area(a, b, c));
}

template<class T>
T get_dist(Point<T> a, Point<T> b) { // 两点间距离
    T dx = a.x - b.x;
    T dy = a.y - b.y;
    return hypot(dx, dy);
}

template<class T>
long long get_dist2(Point<T> p1, Point<T> p2) // 获取距离的平方
{
    long long dx = p1.x - p2.x, dy = p1.y - p2.y;
    return dx * dx + dy * dy;
}


template<class T>
Point<T> rotate(Point<T> a, T b) { // 逆时针旋转b弧度
    return {a.x * cos(b) - a.y * sin(b), a.x * sin(b) + a.y * cos(b)};
}

template<class T>
Point<T> rot90(Point<T> a) // 逆时针90度
{
    return {-a.y, a.x};
}

template<class T>
Point<T> norm(Point<T> a) { //单位向量
    return a / length(a);
}

template<class T>
T get_angle(Point<T> a, Point<T> b) { // 计算向量夹角
    return acos(dot(a,b) / length(a) / length(b));
}

template<class T>
vector<Point<T>> convexHull(vector<Point<T>> p) // 求凸包, 返回凸包上的点
{
    sort(p.begin(), p.end(), [&](P a, P b) -> bool{
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    });
    int n = p.size(), k = 0;
    vector<Point<T>> res(n * 2);
    for (int i = 0; i < n; res[k++] = p[i++])
        while (k > 1 && area(res[k - 2], res[k - 1], p[i]) <= 0)    k--;
    for (int i = n - 2, t = k; i >= 0; res[k++] = p[i--])
        while (k > t && area(res[k - 2], res[k - 1], p[i]) <= 0)    k--;
    res.resize(k - 1);
    return res;
}

template<class T>
T convex_area(vector<Point<T>> p) // 求凸包的面积
{
    double res = 0;
    for (int i = 0; i < p.size(); i++)
        res += cross(p[i], p[(i + 1) % p.size()]);
    return res / 2;
}

template<class T>
Point<T> get_line_intersection(Point<T> p,Point<T> v,Point<T> q,Point<T> w) //获取两直线pv和qw的交点
{
    T a1 = area(q, w, p), a2 = -area(q, w, v);
    return (p * a2 + v * a1) / (a1 + a2);
}

template<class T>
long long rotating_calipers(vector<Point<T>> p) // 旋转卡壳, 获取点对间的最远距离(的平方)
{
    auto q = convexHull(p);
    int n = q.size();
    if(n <= 2)    return get_dist2(q[0],q[1]);

    long long res = 0;
    for(int i = 0,j = 2;i < n;i++)
    {
        auto d = q[i],e = q[i + 1]; //第i个点为 d, 第i+1个点为 e
        while(area(d, e, q[j]) < area(d, e, q[j + 1]))    j = (j + 1) % n;
        res = max(res, max(get_dist2(d, q[j]), get_dist2(e, q[j])));
    }
    return res;
}

template<class T> 
vector<Point<T>> convexCut(const vector<Point<T>> &p, Point<T> q1, Point<T> q2) // 用直线q1q2切凸多边形
{
    vector<Point<T>> res;
    int n = p.size();
    for (int i = 0; i < n; i++)
    {
        Point<T> p1 = p[i], p2 = p[(i + 1) % n];
        int d1 = crossSign(q1, q2, p1), d2 = crossSign(q1, q2, p2);
        if (d1 >= 0)    res.emplace_back(p1);
        if (d1 * d2 < 0)    res.emplace_back(get_line_intersection(p1, p2, q1, q2));
    }
    return res;
}

template<class T>
int contain(vector<Point<T>> ps, Point<T> p) // 点包含(2: inside; 1: on_seg; 0: outside)
{
    int n = ps.size(), res = 0;
    for (int i = 0; i < n; i++)
    {
        auto u = ps[i], v = ps[(i + 1) % n];
        if (on_segment(p, Line<T>(u, v)))   return 1;
        if (cmp(u.y, v.y) <= 0) swap(u, v);
        if (cmp(p.y, u.y) > 0 || cmp(p.y, v.y) <= 0)    continue;
        res ^= crossSign(p, u, v) > 0;
    }
    return res * 2;
}

template<class T>
bool on_line(Point<T> p, Line<T> l) { // 判断点是否在直线上
    return sign(cross(p - l.a, p - l.b)) == 0;
}

template<class T>
bool on_segment(Point<T> p, Line<T> l) { // 判断点是否在线段上
    return on_line(p, l) && sign(dot(p - l.a,p - l.b)) <= 0;// 叉积==0 && 点积<=0
}

template<class T>
T distance_to_line(Point<T> p, Line<T> l) { // 点到直线的距离
    Vector v1 = l.b - l.a, v2 = p - l.a;
    return fabs(cross(v1,v2) / length(v1));
}

template<class T>
T distance_to_segment(Point<T> p, Line<T> l) { // 点到线段的距离
    if(l.a == l.b)	return length(p - l.a);
    Vector v1 = l.b - l.a, v2 = p - l.a, v3 = p - l.b;
    if(sign(dot(v1,v2)) < 0)	return length(v2);
    if(sign(dot(v1,v3)) > 0)	return length(v3);
    return distance_to_line(p, Line<T>{l.a, l.b});
}

template<class T>
Point<T> get_line_projection(Point<T> p, Line<T> l) { // 点在直线上的投影
    Vector v = l.b - l.a;
    return l.a + v * (dot(v,p - l.a) / dot(v, v));
}

template<class T>
bool segment_intersection(Line<T> l1, Line<T> l2) { // 判断两线段是否相交
    T c1 = cross(l1.b - l1.a, l2.a - l1.a), c2 = cross(l1.b - l1.a, l2.b - l1.a);
    T c3 = cross(l2.b - l2.a, l1.b - l2.a), c4 = cross(l2.b - l2.a, l1.a - l2.a);
    return sign(c1) * sign(c2) <= 0 && sign(c3) * sign(c4) <= 0;
}

基础知识

1、前置知识：

（1）π = acos(-1)

（2）余弦定理：$c^2 = a^2 + b^2$ - 2abcos(θ)

2、浮点数比较：

3、向量：

3.1、内积（点积）：A · B = |A| * |B| cos(C)

（1）几何意义：向量A在向量B上的投影与B的长度的乘积。

（2）代码实现：

double dot(Point a,Point b)
{
    return a.x * b.x + a.y * b.y;
}

3.2、外积（叉积）：A × B = |A||B|sin（C）

（1）几何意义：向量A与B张成的平行四边形的有向面积，B在A的逆时针方向为正

（2）代码实现

double cross(Point a,Point b)
{
    return a.x * b.y - b.x * a.y;
}

3.4、常用函数

  • 3.4.1 取模

    double get_length(Point a)
{
    return sqrt(dot(a,a));
}

  • 3.4.2 计算向量夹角

    double get_angle(Point a,Point b)
{
    return acos(dot(a,b) / get_length(a) / get_length(b));
}

  • 3.4.3 计算两个向量构成的平行四边形有向面积

    double area(Point a,Point b,Point c)
{
    return cross(b - a,c - a);
}

  • 3.4.4 向量A顺时针旋转C的角度

    Point rotate(Point a,double angle)
{
    return Point(a.x * cos(angle) + a.y * sin(angle),-a.x * sin(angle) + a.y * cos(angle));
}

    顺时针旋转公式：

$$
(x,y)\begin{pmatrix} cosθ & -sinθ \ sinθ & cosθ \end{pmatrix} \ = （xcosθ + ysinθ，-xsinθ + ycosθ）
$$

逆时针同理，即顺时针旋转-θ角度，将θ换成-θ即可：
$$
(x,y)\begin{pmatrix} cosθ & sinθ \ -sinθ & cosθ \end{pmatrix} \ = （xcosθ - ysinθ，xsinθ + ycosθ）
$$

4、点与线：

  • 4.1 直线定理

    • 一般式：ax + by + c = 0
    • 点向式：p0 + vt
    • 斜截式：y = kx + b

  • 4.2 常用操作

    • 判断点在直线上：A × B = 0（用叉积）

    • 两直线相交

      //cross(v,w) == 0则两直线平行或者重合

      Point get_line_intersection(Point p,vector v,Point q,vector w)
{
    vector u = p - q;
    double t = cross(w,u) / cross(v,w);
    return p + v * t;
}

    • 点到直线距离（右手定则，大拇指向外为正，向内为负）

      double distance_to_line(Point p,Point a,Point b)
{
    vector v1 = b - a,v2 = p - a;
    return fabs(cross(v1,v2) / get_length(v1));
}

    • 点到线段的距离

      double distance_to_segment(Point p,Point a,Point b)
{
    if(a == b)	return get_length(p - a);
    vector v1 = b - a,v2 = p - a,v3 = p - b;
    if(sign(dot(v1,v2)) < 0)	return get_length(v2);
    if(sign(dot(v1,v3)) > 0)	return get_length(v3);
    return distance_to_line(p,a,b);
}

    • 点在直线上的投影

      double get_line_projection(Point p,Point a,Point b)
{
    vector v = b - a;
    return a + v * (dot(v,p - a) / dot(v,v));
}

    • 点是否在线段上

      bool on_segment(Point p,Point a,Point b)
{//叉积为0 && 点积小于等于0
    return sign(cross(p - a,p - b)) == 0 && sign(dot(p - a,p - b)) <= 0;
}

    • 判断两线段是否相交

      bool segment_intersection(Point a1,Point a2,Point b1,Point b2)
{
    double c1 = cross(a2 - a1,b1 - a1),c2 = cross(a2 - a1,b2 - a1);
    double c3 = cross(b2 - b1,a2 - b1),c4 = cross(b2 - b1,a1 - b1);
    return sign(c1) * sign(c2) <= 0 && sign(c3) * sign(c4) <= 0;
}

5、多边形：

​ 5.1 三角形

​ 5.1.1 面积

  • 叉积
    • $S = \frac{\vec{AB} \times \vec{AC}}{2}$
  • 海伦公式
    • $p = \frac{a + b + c}{2}$
    • $S = \sqrt{p(p - a) * (p - b) * (p - c)}$
    • 这个公式乘的次数太多，且开根号，有很大机会会被精度卡掉！！！最后一点讨论海伦精度被卡的题目以及样例。

5.1.2 三角形四心

  • 外心：外接圆圆心
    • 三边中垂线交点。到三角形三个顶点的距离相等。
  • 内心：内切圆圆心
    • 角平分线交点，到三边距离相等。
  • 垂心
    • 三条垂线交点
  • 中心
    • 三条中线交点（到三角形三顶点距离的平方和最小的点，三角形内到三边距离之积最大的点）

template<class T>
Point<T> get_line_intersection(Point<T> p, Vector v, Point<T> q, Vector w) { //获取两直线的交点
    if(!sign(cross(v, w)))    return {INF,INF};//判断共线
    auto u = p - q;
    auto t = cross(w, u) / cross(v, w);
    auto o = p + v * t;
    if(!on_segment(o, Line<T>{p,p + v}) || !on_segment(o, Line<T>{q,q + w})) // 该条件可以获得两个线段的交点
        return {INF,INF};
    return o;
}

            res = min(res, distance_to_segment(p, l));
        }

        cout << res << "\n";
    }
	return 0;
}

数四边形代码：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const double pi = acos(-1),eps = 1e-8;

const int N = 1010;

struct Node
{
    double x,y;
}point[N];
double ans[N * 2];

ll C(ll a,ll b)
{
    ll res = 1;
    for(ll i = a,j = 1;j <= b;j++,i--)
    {
        res = res * i;
        res = res / j;
    }
    return res;
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int n;
    cin >> n;
    for(int i = 0;i < n;i++)    cin >> point[i].x >> point[i].y;
    
    ll cnt = 0;
    for(int i = 0;i < n;i++)
    {
        for(int j = 0;j < n;j++)
        {
            if(i == j)  continue;
            double tmp = atan2(point[j].y - point[i].y,point[j].x - point[i].x);
            if(tmp <= -eps)  tmp += 2 * pi;
            j < i ? ans[j] = tmp : ans[j - 1] = tmp;
        }
        sort(ans,ans + n - 1);
        for(int j = 0;j < n - 1;j++)    ans[j + n - 1] = ans[j] + 2 * pi;
        int k = 1;
        ll tot = 0;
        for(int j = 0;j < n - 1;j++)
        {
            while(fabs(ans[k] - ans[j]) < pi)   k++;
            if(k - j - 1 >= 2)  tot += C(k - j - 1,2);
        }
        cnt += C(n - 1,3) - tot;
    }

    cout << C(n,4) - cnt << "\n";
    
    return 0;
}

struct Line
{
    pdd st,ed;//直线的两个点
}line[N];
pdd pg[N],ans[N];
int q[N];//双端队列

int sign(double x)
{
    if(fabs(x) < eps)   return 0;
    if(x < 0)   return -1;
    return 1;
}

int dcmp(double x,double y)
{
    if(fabs(x - y) < eps)   return 0;
    if(x < y)   return -1;
    return 1;
}

double get_angle(const Line& a)
{
    return atan2(a.ed.y - a.st.y,a.ed.x - a.st.x);//这个参数y在前,x在后
}

pdd operator-(pdd a,pdd b)
{
    return {a.x - b.x,a.y - b.y};
}

double cross(pdd a,pdd b)
{
    return a.x * b.y - a.y * b.x;
}

double area(pdd a,pdd b,pdd c)
{
    return cross(b - a,c - a);
}

bool cmp(const Line& a,const Line& b)
{
    double A = get_angle(a),B = get_angle(b);
    if(!dcmp(A,B))  return area(a.st,a.ed,b.ed) < 0;//如果角度相等,将更靠左的排到前面
    return A < B;
}
//求两直线的交点,点向式
pdd get_line_intersection(pdd p,pdd v,pdd q,pdd w)
{
    auto u = p - q;
    double t = cross(w,u) / cross(v,w);
    return {p.x + v.x * t,p.y + v.y * t};
}

pdd get_line_intersection(Line a,Line b)
{
    return get_line_intersection(a.st,a.ed - a.st,b.st,b.ed - b.st);
}

//bc的交点是否在a的右侧(判断两个直线的交点是不是在一条直线的右边)
bool on_right(Line& a,Line& b,Line& c)
{
    auto o = get_line_intersection(b,c);
    return sign(area(a.st,a.ed,o)) <= 0;
}

double half_plane_intersection()
{
    sort(line,line + cnt,cmp);//将所有直线排序
    int hh = 0,tt = -1;//队头,队尾
    for(int i = 0;i < cnt;i++)
    {
        if(i && !dcmp(get_angle(line[i]),get_angle(line[i - 1])))   continue;//角度相同continue
        while(hh + 1 <= tt && on_right(line[i],line[q[tt - 1]],line[q[tt]]))    tt--;//先判断队尾
        while(hh + 1 <= tt && on_right(line[i],line[q[hh]],line[q[hh + 1]]))    hh++;//再判断队头
        q[++tt] = i;
    }
    while(hh + 1 <= tt && on_right(line[q[hh]],line[q[tt - 1]],line[q[tt]]))    tt--;
    while(hh + 1 <= tt && on_right(line[q[tt]],line[q[hh]],line[q[hh + 1]]))    hh++;

    q[++ tt] = q[hh];
    int k = 0;
    for(int i = hh;i < tt;i++)
        ans[k++] = get_line_intersection(line[q[i]],line[q[i + 1]]);
    double res = 0;
    for(int i = 1;i + 1 < k;i++)
        res += area(ans[0],ans[i],ans[i + 1]);
    return res / 2;
}

{
    int n,m;

    while (n--)
    {
        cin >> m;
        for(int i = 0;i < m;i++)    cin >> pg[i].x >> pg[i].y;
        for(int i = 0;i < m;i++)    
            line[cnt++] = {pg[i],pg[(i + 1) % m]};//起点和终点
    }
    
    cout << fixed << setprecision(3) << half_plane_intersection();
    return 0;
}

三维计算几何基础与三维凸包

#include<iostream>
#include<cmath>
#include<iomanip>

using namespace std;

#define double long double//用long double可以提高精度,但define一下比较好写

const int N = 210;
const double eps = 1e-12;

int n,m;
bool g[N][N];

double rand_eps()
{
    return ((double)rand() / RAND_MAX - 0.5) * eps;
}

struct Point
{
    double x,y,z;
    void shake()//微小扰动
    {
        x += rand_eps(),y += rand_eps(),z += rand_eps();
    }
    Point operator- (Point t)
    {
        return {x - t.x,y - t.y,z - t.z};
    }
    double operator& (Point t)//点积
    {
        return x * t.x + y * t.y + z * t.z;
    }
    Point operator* (Point t)//叉积
    {
        return {y * t.z - t.y * z,z * t.x - x * t.z,x * t.y - y * t.x};
    }
    double len()
    {
        return sqrt(x * x + y * y + z * z);
    }
}q[N];
struct Plane
{
    int v[3];
    Point norm()//法向量
    {
        return (q[v[1]] - q[v[0]]) * (q[v[2]] - q[v[0]]);
    }
    double area()//求面积
    {
        return norm().len() / 2;
    }
    bool above(Point a)
    {
        return ((a - q[v[0]]) & norm()) >= 0;
    }
}plane[N],np[N];

void get_convex_3d()
{
    plane[m++] = {0,1,2};//存储正反两面
    plane[m++] = {2,1,0};
    for(int i = 3;i < n;i++)
    {
        int cnt = 0;
        for(int j = 0;j < m;j++) //检查每一个面
        {
            bool t = plane[j].above(q[i]);
            if(!t)  np[cnt++] = plane[j]; //若照不到则保留
            for(int k = 0;k < 3;k++)
                g[plane[j].v[k]][plane[j].v[(k + 1) % 3]] = t;// 标记每一条边是否被照射
        }
        for(int j = 0;j < m;j++)
            for(int k = 0;k < 3;k++)
            {
                int a = plane[j].v[k],b = plane[j].v[(k + 1) % 3];
                if(g[a][b] && !g[b][a])
                    np[cnt++] = {a,b,i};
            }
        m = cnt;
        for(int j = 0;j < m;j++)    plane[j] = np[j];
    }
}

int main()
{
    cin >> n;
    for(int i = 0;i < n;i++)
    {
        cin >> q[i].x >> q[i].y >> q[i].z;
        q[i].shake();
    }
    get_convex_3d();

    }

    used[0] = false;
    for(int i = n - 1;i >= 0;i--)
    {
        if(used[i]) continue;
        while(top >= 2 && area(q[stk[top - 2]],q[stk[top - 1]],q[i]) <= 0)
            top--;
        stk[top++] = i;
    }
    top--;
}

int rotating_calipers()
{
    if(top <= 2)    return get_dist(q[0],q[n - 1]);

    int res = 0;
    for(int i = 0,j = 2;i < top;i++)
    {//第i个点为d，第i+1个点为e
        auto d = q[stk[i]],e = q[stk[i + 1]];
        while(area(d,e,q[stk[j]]) < area(d,e,q[stk[j + 1]]))    j = (j + 1) % top;
        res = max(res,max(get_dist(d,q[stk[j]]),get_dist(e,q[stk[j]])));
    }
    return res;
}

int main()
{
    cin >> n;
    for(int i = 0;i < n;i++)    cin >> q[i].x >> q[i].y;
    get_convex();
    cout << rotating_calipers();
    return 0;
}

三角剖分

#include<iostream>
#include<iomanip>
#include<cmath>
#include<algorithm>

using namespace std;

#define x first
#define y second

typedef pair<double,double> pdd;

const int N = 55;
const double eps = 1e-8,pi = acos(-1);

double R;
int n;
pdd q[N],r;

int sign(double x)
{
    if(fabs(x) < eps)   return 0;
    if(x < 0)   return -1;
    return 1;
}

int dcmp(double x,double y)
{
    if(fabs(x - y) < eps)   return 0;
    if(x < y)   return -1;
    return 1;
}

pdd operator+(pdd a,pdd b)
{
    return {a.x + b.x,a.y + b.y};
}

pdd operator-(pdd a,pdd b)
{
    return {a.x - b.x,a.y - b.y};
}

pdd operator*(pdd a,double t)
{
    return {a.x * t,a.y * t};
}

pdd operator/(pdd a,double t)
{
    return {a.x / t,a.y / t};
}

double operator*(pdd a,pdd b)
{
    return a.x * b.y - a.y * b.x;
}

double operator&(pdd a,pdd b)
{
    return a.x * b.x + a.y * b.y;
}

double area(pdd a,pdd b,pdd c)
{
    return (b - a) * (c - a);
}

double get_len(pdd a)
{
    return sqrt(a & a);
}

double get_dist(pdd a,pdd b)
{
    return get_len(b - a);
}

double project(pdd a,pdd b,pdd c)
{
    return ((c - a) & (b - a)) / get_len(b - a);
}

pdd rotate(pdd a,double b)
{
    return {a.x * cos(b) + a.y * sin(b),-a.x * sin(b) + a.y * cos(b)};
}

pdd norm(pdd a)
{
    return a / get_len(a);
}

bool on_segment(pdd p,pdd a,pdd b)
{
    return !sign((p - a) * (p - b)) && sign((p - a) & (p - b)) <= 0;
}

pdd get_line_intersection(pdd p,pdd v,pdd q,pdd w)
{
    auto u = p - q;
    auto t = w * u / (v * w);
    return p + v * t;
}

double get_circle_line_intersection(pdd a,pdd b,pdd& pa,pdd& pb)
{
    auto e = get_line_intersection(a,b - a,r,rotate(b - a,pi / 2));
    auto mind = get_dist(r,e);
    if(!on_segment(e,a,b))  mind = min(get_dist(r,a),get_dist(r,b));
    if(dcmp(R,mind) <= 0)   return mind;
    auto len = sqrt(R * R - get_dist(r,e) * get_dist(r,e));
    pa = e + norm(a - b) * len;
    pb = e + norm(b - a) * len;
    return mind;
}

double get_sector(pdd a,pdd b)
{
    auto angle = acos((a & b) / get_len(a) / get_len(b));
    if(sign(a * b) < 0) angle = -angle;
    return R * R * angle / 2;
}

double get_circle_triangle_area(pdd a,pdd b)
{
    auto da = get_dist(r,a),db = get_dist(r,b);
    if(dcmp(R,da) >= 0 && dcmp(R,db) >= 0)  return a * b / 2;
    if(!sign(a * b))    return 0;

    pdd pa,pb;
    auto mind = get_circle_line_intersection(a,b,pa,pb);
    if(dcmp(R,mind) <= 0)   return get_sector(a,b);
    if(dcmp(R,da) >= 0) return a * pb / 2 + get_sector(pb,b);
    if(dcmp(R,db) >= 0) return get_sector(a,pa) + pa * b / 2;
    return get_sector(a,pa) + pa * pb / 2 + get_sector(pb,b);
}

double work()
{
    double res = 0;
    for(int i = 0;i < n;i++)
        res += get_circle_triangle_area(q[i],q[(i + 1) % n]);
    return fabs(res);
}

int main()
{
    cout << fixed << setprecision(2);
    while (cin >> R >> n)
    {
        for(int i = 0;i < n;i++)    cin >> q[i].x >> q[i].y;
        cout << work() << "\n";
    }
    
    return 0;
}

扫描线——三角形面积并

#include<iostream>
#include<algorithm>
#include<vector>
#include<cmath>
#include<iomanip>

using namespace std;

#define x first
#define y second

typedef pair<double,double> pdd;
const int N = 110;
const double eps = 1e-8,INF = 1e6;

int n;
pdd tr[N][3];
pdd q[N];

int sign(double x)
{
    if(fabs(x) < eps)   return 0;
    if(x < 0)   return -1;
    return 1;
}

int dcmp(double x,double y)
{
    if(fabs(x - y) < eps)   return 0;
    if(x < y)   return -1;
    return 1;
}

pdd operator+(pdd a,pdd b)
{
    return {a.x + b.x,a.y + b.y};
}

pdd operator-(pdd a,pdd b)
{
    return {a.x - b.x,a.y - b.y};
}

pdd operator*(pdd a,double t)
{
    return {a.x * t,a.y * t};
}

double operator*(pdd a,pdd b)
{
    return a.x * b.y - a.y * b.x;
}

double operator&(pdd a,pdd b)
{
    return a.x * b.x + a.y * b.y;
}

bool on_segment(pdd p,pdd a,pdd b)
{
    return sign((p - a) & (p - b)) <= 0;
}

pdd get_line_intersection(pdd p,pdd v,pdd q,pdd w)
{
    if(!sign(v * w))    return {INF,INF};//判断共线
    auto u = p - q;
    auto t = w * u / (v * w);
    auto o = p + v * t;
    if(!on_segment(o,p,p + v) || !on_segment(o,q,q + w))
        return {INF,INF};
    return o;
}

double line_area(double a,int side)
{
    int cnt = 0;
    for(int i = 0;i < n;i++)
    {
        auto t = tr[i];
        if(dcmp(t[0].x,a) > 0 || dcmp(t[2].x,a) < 0)    continue;
        if(!dcmp(t[0].x,a) && !dcmp(t[1].x,a))
        {
            if(side)    q[cnt++] = {t[0].y,t[1].y};
        }
        else if(!dcmp(t[2].x,a) && !dcmp(t[1].x,a))
        {
            if(!side)   q[cnt++] = {t[2].y,t[1].y};
        }
        else
        {
            double d[3];
            int u = 0;
            for(int j = 0;j < 3;j++)
            {
                auto o = get_line_intersection(t[j],t[(j + 1) % 3] - t[j],{a,-INF},
                {0,INF * 2});
                if(dcmp(o.x,INF))
                    d[u++] = o.y;
            }
            if(u)
            {
                sort(d,d + u);
                q[cnt++] = {d[0],d[u - 1]};
            }
        }
    }
    if(!cnt)    return 0;
    for(int i = 0;i < cnt;i++)
        if(q[i].x > q[i].y)
            swap(q[i].x,q[i].y);
    sort(q,q + cnt);
    double res = 0,st = q[0].x,ed = q[0].y;
    for (int i = 1; i < cnt; i++)
        if(q[i].x <= ed)    ed = max(ed,q[i].y);
        else
        {
            res += ed - st;
            st = q[i].x,ed = q[i].y;
        }
    res += ed - st;
    return res;
}

double range_area(double a,double b)
{
    return (line_area(a,1) + line_area(b,0)) * (b - a) / 2;
}

int main()
{
    cin >> n;
    vector<double> xs;
    for(int i = 0;i < n;i++)
    {
        for(int j = 0;j < 3;j++)
        {
            cin >> tr[i][j].x >> tr[i][j].y;
            xs.push_back(tr[i][j].x);
        }
        sort(tr[i],tr[i] + 3);
    }
    for(int i = 0;i < n;i++)
        for(int j = i + 1;j < n;j++)
            for(int x = 0;x < 3;x++)
                for(int y = 0;y < 3;y++)
                {
                    auto o = get_line_intersection(tr[i][x],tr[i][(x + 1) % 3] - tr[i][x],
                    tr[j][y],tr[j][(y + 1) % 3] - tr[j][y]);
                    if(dcmp(o.x,INF))   xs.push_back(o.x);
                }
    sort(xs.begin(),xs.end());
    double res = 0;
    for(int i = 0;i + 1 < xs.size();i++)
        if(dcmp(xs[i],xs[i + 1]))
            res += range_area(xs[i],xs[i + 1]);

    cout << fixed << setprecision(2) << res;
    return 0;
}

自适应辛普森积分

#include<iostream>
#include<iomanip>
#include<cmath>

using namespace std;

const double eps = 1e-12;

double f(double x)
{
    return sin(x) / x;
}

double simpson(double l,double r)//将l~r这一段当抛物线求面积是多少
{
    auto mid = (l + r) / 2;
    return (r - l) * (f(l) + 4 * f(mid) + f(r)) / 6;
}

double asr(double l,double r,double s)//自适应的递归函数
{
    auto mid = (l + r) / 2;
    auto left = simpson(l,mid),right = simpson(mid,r);
    if(fabs(left + right - s) < eps)    return left + right;
    return asr(l,mid,left) + asr(mid,r,right); 
}

int main()
{
    double l,r;
    cin >> l >> r;

    cout << fixed << setprecision(6) << asr(l,r,simpson(l,r));
    return 0;
}

海伦公式（long double和 lsqrt写法）：

#include<bits/stdc++.h>

using namespace std;

using pii = pair<int, int>;

bool check(pii a, pii b, pii c)
{
	int dy1 = b.second - a.second, dx1 = b.first - a.first;
	int dy2 = c.second - b.second, dx2 = c.first - b.first;
	return !(dy1 * dx2 == dy2 * dx1);
}

long double get_dist(pii a, pii b)
{
	int dx = a.first - b.first, dy = a.second - b.second;
	return sqrtl(dx * dx + dy * dy);
}

long double get_area(pii a, pii b, pii c)
{
	long double d1 = get_dist(a, b), d2 = get_dist(b, c), d3 = get_dist(a, c);
	long double p = (d1 + d2 + d3) / 2;
	return sqrtl(p * (p - d1) * (p - d2) * (p - d3));
}

void solve()
{
	int n;
	cin >> n;

	vector<pii> point(n + 1);
	for (int i = 1; i <= n; i++)	
	{
		cin >> point[i].first >> point[i].second;
	}

	long double res = 1e18;

	for (int i = 1; i <= n; i++)
	{
		for (int j = i + 1; j <= n; j++)
		{
			for (int k = j + 1; k <= n; k++)
			{
				if (!check(point[i], point[j], point[k]))	continue;

				res = min(res, get_area(point[i], point[j], point[k]));
			}
		}
	}

	if (res > 1e17)	cout << "-1\n";
	else	cout << fixed << setprecision(8) << res << "\n";
}

int main()
{
	cin.tie(nullptr)->sync_with_stdio(false);

	int t;
	cin >> t;
	
	while (t--)
	{
		solve();
	}
	
	return 0;
}
