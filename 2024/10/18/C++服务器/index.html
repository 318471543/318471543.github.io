TCP服务器（Linux）

简易的，不使用任何线程等优化的，便于理解。

#include <iostream>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <fcntl.h>
#include <errno.h>
#include <string.h>

int main()
{
	// 1、创建socket套接字
	int listen_socket = socket(AF_INET, SOCK_STREAM, 0);

	if (-1 == listen_socket)
	{
		std::cout << "create listen_socket failed! errcode: %m" << std::endl;
		return -1;
	}

	// 2、给这个socket绑定一个端口
	struct sockaddr_in local = { 0 };
	local.sin_family = AF_INET;
	local.sin_port = htons(8080); // 大小端的问题,将主机字节序转换为网络字节序
	// 服务端 选项 网卡 127.0.0.1（本地环回） 只接受哪个网卡的数据 一般写全0地址表示全部都接收 
	local.sin_addr.s_addr = inet_addr("0.0.0.0"); // 字符串IP地址转换成整数IP

	if (-1 == bind(listen_socket, (struct sockaddr*)&local, sizeof(local)))
	{
		std::cout << "bind listen socket failed !!! errcode: %m" << std::endl;
		return -1;
	}
	// 3、给这个socket开启监听属性
	if (-1 == listen(listen_socket, 10))
	{
		std::cout << "start listen failed !!! errcode: %m" << std::endl;
		return -1;
	}
	// 4、等待客户端连接
	while (true)
	{
		int client_socket = accept(listen_socket, NULL, NULL);
		if (-1 == client_socket)
			continue;
		std::cout << "new connect, socket: " << client_socket << std::endl;

		while (true)
        {
            // 5、开始通讯
            char buffer[1024] = { 0 };
            int ret = recv(client_socket, buffer, 1024, 0);
            if (ret <= 0)	break;
            std::cout << client_socket << ": " << buffer << std::endl;
            send(client_socket, buffer, strlen(buffer), 0);
        }

        std::cout << "socket: " << client_socket << ", disconnet." << std::endl;

        // 6、关闭连接
        close(client_socket);
	}

	return 0;
}

效果展示：

用

telnet 127.0.0.1 8080

或者

nc 127.0.0.1 8080

即可与服务器进行通信。

而curl

curl 127.0.0.1:8080

输出http的一些信息。

问了AI，回答是：

当 curl 发送了这个请求之后，服务器需要处理这个请求，并返回一个 HTTP 响应。如果服务器是基于 Web 的 HTTP 服务器，它可能会返回一个 HTML 页面。如果服务器是一个简单的 TCP 服务器（不是 HTTP 服务器），它可能无法理解 HTTP 请求，并且可能不会有任何响应，或者返回一个错误。

UDP服务器（Linux）

#include <iostream>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#include <cstring>
#include <cerrno>

int main() {
    // 创建 UDP 套接字
    int sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    if (sockfd < 0) {
        std::cerr << "Can't create socket: " << strerror(errno) << std::endl;
        return -1;
    }

    struct sockaddr_in local = { 0 };
    local.sin_family = AF_INET;          // 使用 IPv4 地址
    local.sin_port = htons(8080);        // 端口号
    local.sin_addr.s_addr = inet_addr("0.0.0.0"); // 字符串IP地址转换成整数IP

    // 绑定套接字到指定端口
    if (bind(sockfd, (const struct sockaddr*)&local, sizeof(local)) < 0) {
        std::cerr << "Bind failed: " << strerror(errno) << std::endl;
        return -1;
    }

    struct sockaddr_in clientAddr = { 0 };
    socklen_t clientAddrLen = sizeof(clientAddr);
    while (true) {
        char buffer[1024] = { 0 };
        // 接收客户端数据
        int n = recvfrom(sockfd, buffer, sizeof(buffer) - 1, 0, 
                         (struct sockaddr*)&clientAddr, &clientAddrLen);
        if (n < 0) {
            std::cerr << "Recvfrom failed: " << strerror(errno) << std::endl;
            break;
        }
        buffer[n] = '\0';  // 确保字符串结束

        // std::cout << "Received: " << buffer << std::endl;
        // 输出收到的消息和客户端信息
        std::cout << "Received from IP: "
                  << inet_ntoa(clientAddr.sin_addr)
                  << " Port: " << ntohs(clientAddr.sin_port)
                  << ": " << buffer << std::endl;

        // 发送数据回客户端
        if (sendto(sockfd, buffer, n, 0, 
                   (const struct sockaddr*)&clientAddr, clientAddrLen) < 0) {
            std::cerr << "Sendto failed: " << strerror(errno) << std::endl;
            break;
        }
    }

    // 关闭套接字
    close(sockfd);
    return 0;
}

效果展示：

nc -u 127.0.0.1 8080

不能再用telnet了（解释如下）：

telnet 是一个命令行界面的网络协议客户端，它主要用于连接到 TCP 服务器，因为它仅支持基于 TCP 的协议。

同理，curl就更不行了。

测试

TCP、UDP可以使用同一个端口吗？

答案是可以的，下面给出实践：

查看端口使用情况：

ss -tulnp | grep 8080
  • -t (TCP) 仅显示 TCP 协议的连接。
  • -u (UDP) 仅显示 UDP 协议的连接。
  • -l (listen) 仅显示监听状态的套接字。
  • -n (numeric) 不解析地址和端口号，直接显示 IP 地址和端口号。
  • -p (program) 显示连接到套接字的程序名和进程ID。
