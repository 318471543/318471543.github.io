1002 City Upgrading

题意：

​ 有一个树，树上每个点有一个权值。选择树上的一个结点并花费一个权值后，会将该结点的信息传给邻边。问将信息传递给整个树的最小权值是多少？

思路：

​ 看到这道题目，很容易想到树形DP的模板题：没有上司的舞会。

  1. DP状态定义如下：

    • $f[u][num]$：以 $u$ 为根的子树信息被全部传递，且当前结点 $u$ 的状态为 $num$ 的集合（ $num$ 取值：0,1,2）。
      • $num = 0$：当前结点 $u$ 信息被父节点传递。
      • $num = 1$：当前结点 $u$ 信息被自己传递。
      • $num = 2$：当前结点 $u$ 信息被子结点传递。

  2. 分析转移公式：

    • $f[u][0] += min(f[v][1],f[v][2])$
    • $f[u][1] += min(f[v][0] - a[u],f[v][1],f[v]][2])$
    • $f[u][2] += min(f[v][1],f[v][2])$，其中至少有一项 $f[v][1]$，毕竟定义要求该结点 $u$ 是要被孩子激活的。

收获：

除了收获了这道题以外，我还弄清楚了之前的一个疑惑：树本身是无向的，而一但确定根后，就变成有向的了（建有向图），那么进而还要寻找谁是根节点；而如果直接建无向图（即建正反边），那么谁是根节点都可以。

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

void solve()
{
    int n;
    cin >> n;

    vector<int> a(n + 1);
    for(int i = 1;i <= n;i++)   cin >> a[i];

    vector<vector<int>> g(n + 1);
    vector<int> has_father(n + 1);
    for(int i = 0;i < n - 1;i++)
    {
        int a,b;
        cin >> a >> b;
        g[a].push_back(b);
        has_father[b] = a;
    }

    int root = 1;
    while(has_father[root]) root = has_father[root];

    vector<vector<ll>> f(n + 1,vector<ll>(4));
    //f[u][0]:父亲激活自己 f[u][0] += min(f[j][1],f[j][2])
    //f[u][1]:自己激活自己 f[u][1] += min(f[j][0] - a[u],f[j][1],f[j][2])
    //f[u][2]:儿子激活自己 f[u][2] = min(f[u][1],f[u][2]),至少1个f[u][1]
    //f[u][?]:以u为根的子树中，全部接通信号，且u点状态为0,1,2的所有集合
    function<void(int)> dfs = [&](int u) -> void
    {
        f[u][0] = a[has_father[u]];
        f[u][1] = a[u];
        vector<int> vec;
        ll sum = 0;
        for(auto v : g[u])
        {
            dfs(v);
            f[u][0] += min({f[v][1],f[v][2]});
            f[u][1] += min({f[v][0] - a[u],f[v][1],f[v][2]});
            vec.push_back(v);
            sum += min(f[v][1],f[v][2]);
        }

        ll res = 1e18;
        for(int i = 0;i < vec.size();i++)
        {
            int id = vec[i];
            ll t = sum - min(f[id][1],f[id][2]);
            t +=  f[id][1];
            res = min(res,t);
        }

        f[u][2] = res;
    };

    dfs(root);

    cout << min(f[root][1],f[root][2]) << "\n";
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int t;
    cin >> t;

    while(t--)  solve();

    return 0;
}

1005 Cyclically Isomorphic

题意：

​ 给定 $n$ 个字符串，长度均为 $m$，随后 $q$ 个询问，每次询问两个字符串是否是“循环同构的”？

思路：

​ 比较容易想到一个 $O(n)$ 算法求最小表示法的一个模板题：项链

方法一：

#include<bits/stdc++.h>

using namespace std;

const int N = 2e6 + 10;

int n;
char a[N];

int get_min(char s[])
{
    int i = 0,j = 1;
    while (i < n && j < n)
    {
        int k = 0;
        while(k < n && s[i + k] == s[j + k])  k++;
        if(k == n)  break;
        if(s[i + k] > s[j + k]) i += k + 1;
        else  j += k + 1;
        if(i == j)  j++;
    }
    int k = min(i,j);
    s[k + n] = 0;
    return k;
}

void solve()
{
    int cnt,m;
    cin >> cnt >> n;
    vector<string> s(cnt + 1);
    for(int i = 1;i <= cnt;i++)
    {
        cin >> a;
        memcpy(a + n,a,n);
        int x = get_min(a);
        s[i] = a + x;
    }
    
    cin >> m;
    while (m--)
    {
        int a,b;
        cin >> a >> b;
        cout << (s[a] == s[b] ? "Yes\n" : "No\n");
    }
    
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int t;
    cin >> t;

    while(t--)  solve();

    return 0;
}

方法二：

#include<bits/stdc++.h>

using namespace std;

const int N = 2e5 + 10;

int tot,last;
struct Node
{
    int len,fa;
    int ch[26];
    void init()
    {
        len = fa = 0;
        memset(ch,0,sizeof(ch));
    }
}node[N];

void extend(int c)
{
    int p = last,np = last = ++tot;
    node[np].len = node[p].len + 1;
    for(;p && !node[p].ch[c];p = node[p].fa)
        node[p].ch[c] = np;
    if(!p)  node[np].fa = 1;
    else
    {
        int q = node[p].ch[c];
        if(node[q].len == node[p].len + 1)   node[np].fa = q;
        else
        {
            int nq = ++ tot;
            node[nq] = node[q],node[nq].len = node[p].len + 1;
            node[q].fa = node[np].fa = nq;
            for(;p && node[p].ch[c] == q;p = node[p].fa)    node[p].ch[c] = nq;
        }
    }
}

string get(string s)
{
    for(int i = 1;i <= tot;i++) node[i].init();
    tot = last = 1;
    
    for(auto c : s) extend(c - 'a');
    for(auto c : s) extend(c - 'a');

    string res;
    int p = 1;
    for(int i = 0;i < s.size();i++)
    {
        for(int j = 0;j < 26;j++)
            if(node[p].ch[j])
            {
                p = node[p].ch[j];
                res += char(j + 'a');
                break;
            }
    }
    return res;
}

void solve()
{
    int n,m;
    cin >> n >> m;

    vector<string> str(n + 1);
    for(int i = 1;i <= n;i++)
    {
        string s;
        cin >> s;
        str[i] = get(s);
    }

    int q;
    cin >> q;

    while (q--)
    {
        int a,b;
        cin >> a >> b;

        cout << (str[a] == str[b] ? "Yes\n" : "No\n");
    }
    
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int t;
    cin >> t;

    while(t--)  solve();

    return 0;
}

1009 Assertion

题意：

​ 将 $m$ 分成 $n$ 份，是否至少有一份的数 $\ge$ d，如果是输出“Yes”，否则输出“No”。

思路：

​ 先均分（即 $\frac{m}{n}$ 个），如果有余数的话则加 $1$，然后判断这个值是否 $\ge$ d 即可。

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int,int> pii;

#define IOS ios::sync_with_stdio(false),cin.tie(nullptr),cout.tie(nullptr)
#define cf(_) int _;cin >> _;while(_--)

int main()
{
    IOS;
    cf(_)
    {
        int n,m,d;
        cin >> n >> m >> d;

        if(((m - n) % n != 0) + m / n >= d)   cout << "Yes\n";
        else    cout << "No\n";
    }
    return 0;
}
