前缀和

一维前缀和

求区间[l,r]之间的和：$s[r] - s[l - 1]$

核心代码：

for(int i = 1; i<=n;i++)     s[i] = s[i-1]+ a[i];

一维异或和

求区间 $[l,r]$ 之间的异或和：$s[r]$ ^ $s[l - 1]$

核心代码：

for(int i = 1;i <= n;i++)	sum[i] = sum[i - 1] ^ a[i];

二维前缀和

模板：

auto init = [&]() -> void // 初始化
{
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + a[i][j];
};

auto get = [&](int x1, int y1, int x2, int y2) -> ll // 返回矩阵[x1, y1]~[x2, y2]的和
{
    return s[x2][y2] - s[x1 - 1][y2] - s[x2][y1 - 1] + s[x1 - 1][y1 - 1];
};

模板题代码：

#include<bits/stdc++.h>

using namespace std;

const int N = 1010;

int main()
{
    int n,m,q;

    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= m; j++)
        {
            scanf("%d",&a[i][j]);
            s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + a[i][j];
        }
    }
    while (q--)
    {
        int x1,y1,x2,y2;
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
        printf("%d\n",s[x2][y2] - s[x1 - 1][y2] - s[x2][y1 - 1] + s[x1 - 1][y1 - 1]);
    }
    return 0;
}

一维差分

核心代码：

在区间 $[l,r]$ 加上一个数 $c$：$b[l] += c, b[r + 1] -= c;$

模板：

auto insert = [&](int l, int r, int c) -> void // 预处理
{
    b[l] += c;
    b[r + 1] -= c;
};

for (int i = 1; i <= n; i++)    b[i] += b[i - 1]; // 求一遍前缀和即为现数组

模板题：差分

二维差分

模板：

auto insert = [&](int x1, int y1, int x2, int y2, int c) -> void // 预处理
{
    b[x1][y1] += c;
    b[x1][y2+1] -= c;
    b[x2+1][y1] -= c;
    b[x2+1][y2+1] += c;
};

for(int i = 1;i <= n;i++) // 求一遍前缀和,类似于二维前缀和的预处理操作,b[i][j] 即为现数组
    for(int j = 1;j <= m;j++)
        b[i][j] += b[i-1][j] + b[i][j-1] - b[i-1][j-1];

模板题代码：

#include<bits/stdc++.h>

using namespace std;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    cout.tie(nullptr);
    
    int n, m, q;
    cin >> n >> m >> q;

    vector<vector<int>> b(n + 2, vector<int>(m + 2));
    auto insert = [&](int x1, int y1, int x2, int y2, int c) -> void // 预处理
    {
        b[x1][y1] += c;
        b[x1][y2+1] -= c;
        b[x2+1][y1] -= c;
        b[x2+1][y2+1] += c;
    };

    for(int i = 1;i <= n;i++)
        for(int j = 1, x;j <= m;j++)
            cin >> x, insert(i, j, i, j, x);

    while (q--)
    {
        int x1, y1, x2, y2, c;
        cin >> x1 >> y1 >> x2 >> y2 >> c;
        insert(x1, y1, x2, y2, c);
    }

    for(int i = 1;i <= n;i++) // 求一遍前缀和,类似于二维前缀和的预处理操作,b[i][j] 即为现数组
        for(int j = 1;j <= m;j++)
            b[i][j] += b[i-1][j] + b[i][j-1] - b[i-1][j-1];

    for(int i = 1;i <= n;i++)
        for(int j = 1;j <= m;j++)
            cout << b[i][j] << " \n"[j == m];
    return 0;
}

推荐题单：

二分 + 双指针 / 二分套二分：A. Ntarsis’ Set

$ 2023CCPC $桂林站二分签到题：M. Flipping Cards

细谈CCPC二分签到：

很签吗？？？不签啊！但从各方面来说，但经典卧槽！

这题让我魂牵梦绕了好久好久。。。

接下来，我们将细细探讨：

  1. 二分答案
    • 二分答案好说，但永远的问题就是如何 $check$ ？好，下面来谈！
  2. 可以想到，如果可以保证 $\geq mid$ 的数的个数有 $\frac{n + 1}{2}$ 个，那么由于是 $\geq$ 符号，则第 $\frac{n+1}{2}$ 个一定就是答案。 而如果个数更多，即超过 $\frac{n+1}{2}$ 个，那么答案要么是 $mid$，要么就是更靠后的答案（即令 $l = mid$）。
    • 那么怎么计算这样的数有多少个呢？毕竟还有连续区间翻转的情况？那好，下面谈！
  3. 不翻转的情况下，我们统计 $a$ 数组中有多少个数 $\geq mid$，然后不够的话就得需要通过翻转来凑！怎么凑呢？
    • 如果该位置的 $a_i < mid$，而 $b_i >= mid$，那么交换之后会贡献 $1$ 个答案；
    • 如果该位置的 $a_i$ 和 $b_i$ 同时 $\geq$ 或 $< mid$ ，则贡献为 $0$；
    • 如果该位置的 $a_i \geq mid$，而 $b_i < mid$，则交换后贡献 $-1$；
  4. 因此，有了这种转换的想法后，我们只需看最大连续子段和是多少就可以知道翻转后能得到的 $1$ 贡献最多是多少了！
  5. 因此，最终判断只需看 $cnt_{初始1的个数} + cnt_{翻转后提供1贡献} \geq \frac{n+1}{2}$ 即可。
#include<bits/stdc++.h>

using namespace std;

#define pii pair<int, int>

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int n;
    cin >> n;

    vector<pii> a(n + 1);
    for (int i = 1; i <= n; i++)    cin >> a[i].first >> a[i].second;

    auto check = [&](int mid) -> int
    {
        vector<int> v(n + 1);
        for (int i = 1; i <= n; i++)
        {
            v[i] = (a[i].second >= mid) - (a[i].first >= mid);
        }

        int res = 0, cur = 0, cnt = 0;
        for (int i = 1; i <= n; i++)
        {
            if (a[i].first >= mid)  cnt++;
            cur = max(v[i], cur + v[i]);
            res = max(res, cur);
        }

        if (res + cnt >= (n + 1) / 2)    return true;
        return false;
    };

    int l = 1, r = 1e9;
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (check(mid))   l = mid;
        else    r = mid - 1;
    }
    
    cout << l << "\n";
    return 0;
}

三分

三分时间复杂度：O（n$^2$）

三分与二分的思想类似，也是通过不断缩小答案可能的区间范围，来找到最终答案所在的位置。

三分一般被用来查找某类单峰函数的峰值。

三分需要找三等分点：M1 = (R - L) / 3 + L,M2 = (R - L) / 3 + 2L。

注意：

但也有某类单峰函数不能用三分！

某些问题的函数可能是单峰的，但峰值两边并非严格递增（递减）的，即有可能出现”平台“（）一条与x轴平行的直线。

什么时候使用三分？

​ 即一个函数，在考虑的范围[l,r]内是单峰的，并且从峰值往左右两边都是要严格单调递增或严格单调递减的。

整数三分

$while(l + 2 < r)$ 需要区间可以分成三段！

最后暴力枚举 $l$ ~ $r$ 之间的结果即可！

整数三分凸函数模板：

int l = 0,r = 1e9;
while (l + 2 < r)
{
    int m1 = l + (r - l) / 3;
    int m2 = l + (r - l) / 3 * 2;
    if(f(m1) > f(m2))   r = m2;
    else    l = m1;
}

ll res = l;
for(int i = l + 1; i <= r;i ++)
    if(f(i) > f(res))   res = i;//遍历一遍，获取最终答案

整数三分凹函数模板：

int l = 0,r = 1e9;
while (l + 2 < r)
{
    int m1 = l + (r - l) / 3;
    int m2 = l + (r - l) / 3 * 2;
    if(f(m1) < f(m2))   r = m2;
    else    l = m1;
}

ll res = f(l);
for(int i = l + 1;i <= r;i++)   res = min(res,f(i));//遍历一遍，获取最终答案

推荐题代码：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 1e5 + 10;

int n,A,R,M;
int a[N];

ll f(int x)
{
    ll y = 0,z = 0;
    for(int i = 1;i <= n;i++)
    {
        if(a[i] < x)    y += x - a[i];
        else    z += a[i] - x;
    }

    ll res = min(y,z) * M; 
    if(y > z)   res += (y - z) * A;
    else    res += (z - y) * R;
    return res;
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    cin >> n >> A >> R >> M;
    M = min(M,A + R);
    for(int i = 1;i <= n;i++)   cin >> a[i];

    int l = 0,r = 1e9;
    while (l + 2 < r)
    {
        int m1 = (r - l) / 3 + l;
        int m2 = (r - l) / 3 * 2 + l;
        if(f(m1) > f(m2))   l = m1;
        else    r = m2;
    }
    
    ll res = f(l);
    for(int i = l + 1;i <= r;i++)   res = min(res,f(i));

    cout << res << "\n";
    return 0;
}

浮点数三分

【模板】三分法

模板记忆口诀：哪个情况更糟就删去哪一部分（每次会删去三分之一的区间）：

比如要求最大值，就要把小的那一部分给删去；

要求最小值，则把大的那一部分删去。

浮点三分凸函数模板：

auto f = [&](double x) -> double
{
    double res = 0;
    for(int i = 1;i <= n;i++)
        res += a[i] * pow(x,n - i + 1);
    return res;
};

while (l + eps < r)
{
    double m1 = (r - l) / 3 + l;
    double m2 = (r - l) / 3 * 2 + l;
    if(f(m1) > f(m2))    r = m2;
    else    l = m1;
}
cout << fixed << setprecision(5) << l << "\n";

模板题代码：

#include<bits/stdc++.h>

using namespace std;

constexpr double eps = 1e-7;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int n;
    double l,r;
    cin >> n >> l >> r;

    vector<double> a(n + 1);
    for(int i = 1;i <= n;i++)   cin >> a[i];
    double c;
    cin >> c;

    auto f = [&](double x) -> double
    {
        double res = 0;
        for(int i = 1;i <= n;i++)
            res += a[i] * pow(x,n - i + 1);
        return res + c;
    };

    while (l + eps < r)
    {
        double m1 = (r - l) / 3 + l;
        double m2 = (r - l) / 3 * 2 + l;
        if(f(m1) > f(m2))    r = m2;
        else    l = m1;
    }
    
    cout << fixed << setprecision(5) << l << "\n";
    return 0;
}
