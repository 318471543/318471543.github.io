背包九讲

01背包

有 N 个物品和一个容量为 V 的背包，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N,V $\le$ 1000

0 < $v_i,w_i$ $\le$ 1000

二维数组（朴素版）：

状态f[i] [j] 定义为：前 i 个物品，背包容量 j 下的最优解（最大价值）。

有 N 件物品，则需要 N 次决策，每一次对第 i 件物品的决策，状态f[i] [j]不断由之前的状态更新而来。

1、当前背包容量不够（j < v[i]），没得选，因此前 i 个物品最优解即为前 i−1个物品最优解：

对应代码：f[i] [j] = f[i -1] [j]。

2、当前背包容量够，可以选，因此需要决策选与不选第 i 个物品：

选：f[i] [j] = f[i - 1] [j - v[i]] + w[i]。
不选：f[i] [j] = f[i - 1] [j] 。

我们的决策是如何取到最大价值，因此以上两种情况取 max() 。

#include<bits/stdc++.h>
using namespace std;
const int N = 1010;
int n,m;
int v[N],w[N];//v表示体积，w表示价值
int f[N][N];//f表示状态
int main()
{
    cin >> n >> m;//物品个数、背包容量
    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];//输入每个物品的体积、价值
    for(int i = 1;i <= n;i++)//f[0][0~m]表示考虑0件物品，总体积不超过0~m，故最大价值为0，全局变量以默认初始化，因此从1开始
        for(int j = 0;j <= m;j++)
        {
            f[i][j] = f[i-1][j];
            if(j >= v[i])   f[i][j] = max(f[i][j],f[i-1][j-v[i]] + w[i]);
        }
    cout << f[n][m] << endl;
    return 0;
}

一维数组（优化版）：

状态f[j]定义：N 件物品，背包容量j下的最优解。

注意枚举背包容量j必须从m开始。

为什么一维情况下枚举背包容量需要逆序？在二维情况下，状态f[i] [j]是由上一轮i - 1的状态得来的，f[i] [j]与f[i - 1] [j]是独立的。而优化到一维后，如果我们还是正序，则有f[较小体积]更新到f[较大体积]，则有可能本应该用第i-1轮的状态却用的是第i轮的状态（因为体积小的状态已经被更新了，当体积大的需要用到体积小的去更新时，用到的是第i轮，而非第i - 1轮）。

例如，一维状态第i轮对体积为 3 的物品进行决策，则f[7]由f[4]更新而来，这里的f[4]正确应该是f[i - 1] [4]，但从小到大枚举j这里的f[4]在第i轮计算却变成了f[i] [4]。当逆序枚举背包容量j时，我们求f[7]同样由f[4]更新，但由于是逆序，这里的f[4]还没有在第i轮计算，所以此时实际计算的f[4]仍然是f[i - 1] [4]。

故状态转移方程为：f[j] = max(f[j], f[j - v[i]] + w[i] 。

#include<bits/stdc++.h>

using namespace std;

const int N = 1010;
int n,m;
int v[N],w[N];//v表示体积，w表示价值
int f[N];//一维数组

int main()
{
    cin >> n >> m;//物品个数、背包容量
    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];//输入每个物品的体积、价值
    
    for(int i = 1;i <= n;i++)
        for(int j = m;j >= v[i];j--)//逆序更新
        {
            f[j] = max(f[j],f[j-v[i]] + w[i]);
        }
    cout << f[m] << endl;
    return 0;
}

完全背包

有 n 种物品和一个容量为 v 的背包，每个物品有无限个，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N,V $\le$ 1000

0 < $v_i,w_i$ $\le$ 1000

二维数组（朴素版）：

与01背包的代码几乎完全一样，只不过枚举体积的顺序是从小到大开始！

公式推导过程：

f[i] [j] = max(f[i - 1] [j],f[i - 1] [j - v] + w,f[i - 1] [j - 2v] + 2w,f[i - 1] [j - 3v] + 3w…)

然后突然有了一个“神奇”的表达式，f[i] [j - v]如下：

f[i] [j - v] = max(f[i - 1] [j - v],f[i - 1] [j - 2v] + w,f[i - 1] [j - 3v] + 2w,…)

不难看出，标亮的式子中：上面比下面多加了一个w。

因此：f[i] [j] = max(f[i - 1] [j],f[i] [j - v] + w)。

#include<iostream>

using namespace std;

const int N = 1010;
int n,m;
int f[N][N];
int v[N],w[N];

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m;

    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];

    for(int i = 1;i <= n;i++)
    {
        for(int j = 0;j <= m;j++)
        {
            f[i][j] = f[i - 1][j];
            if(j >= v[i])   f[i][j] = max(f[i][j],f[i][j - v[i]] + w[i]);
        }
    }

    cout << f[n][m] << "\n";
    return 0;
}

一维数组（优化版）：

如果与01背包进行对比：
f[i] [j] = max(f[i - 1] [j],f[i - 1] [j - v] + w) 01背包

f[i] [j] = max(f[i - 1] [j],f[i] [j - v] + w) 完全背包

可以发现，完全背包需要用到当前的状态，即从小到大枚举。

#include<bits/stdc++.h>

using namespace std;

const int N = 1010;
int n,m;
int f[N],v[N],w[N];

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m;

    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];

    for(int i = 1;i <= n;i++)
        for(int j = v[i];j <= m;j++)//体积从小到大枚举
        {
            f[j] = max(f[j],f[j - v[i]] + w[i]);
        }
        
    cout << f[m];
    return 0;
}

多重背包

多重背包I

有 N 种物品和一个容量为 V 的背包，每种物品最多 s 件，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N,V $\le$ 100

0 < $v_i,w_i,s_i$ $\le$ 100

朴素版：

#include<bits/stdc++.h>
using namespace std;
const int N = 110;
int n,m;
int v[N],w[N],s[N];
int f[N][N];
int main()
{
    cin >> n >> m;
    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i] >> s[i];
    for(int i = 1;i <= n;i++)
        for(int j = 0;j <= m;j++)
            for(int k = 0;k <= s[i] && k * v[i] <= j;k++)
                f[i][j] = max(f[i][j],f[i-1][j-v[i]*k] + w[i] * k);
    cout << f[n][m] << endl;
    return 0;
}

多重背包II

有 N 种物品和一个容量为 V 的背包，每种物品最多 s 件，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N $\le$ 1000

0 < V $\le$ 2000

0 < $v_i,w_i,s_i$ $\le$ 2000

多重背包的二进制优化：

利用二进制优化，时间复杂度可以从 O(N M S) 降到O(N logS M ),从4∗10^9^降到了2∗10^7^。

思想：将每种物品拆分成1,2,4,8,16,32…，即分成1,2,4，…，2^k^,s这些个物品，其中s小于2^k+1^。这样可以利用01背包的思想，拼凑出任意数量的解（包括最优解），然后调用01背包模板即可。

举例：如果某种物品数量s = 200，则可拆分成：1,2,4,8,16,32,64,73（73 < 128）。

#include<bits/stdc++.h>
using namespace std;
const int N = 12010,M = 2010;//一共有1000个物品，每个物品最多拆分成log2000个物品，故是N的由来
int n,m;
int v[N],w[N];
int f[M];
int main()
{
    cin >> n >> m;
    int cnt = 0;
    for (int i = 1; i <= n; i++)
    {
        int a,b,s;
        cin >> a >> b >> s;
        int k = 1;
        while (k <= s)
        {
            cnt++;
            v[cnt] = a * k;
            w[cnt] = b * k;
            s -= k;
            k *= 2;
        }
        if(s > 0)
        {
            cnt++;
            v[cnt] = a * s;
            w[cnt] = b * s;
        }
    }
    n = cnt;//每个数选与不选都能拼凑出任意的数字，这就是2进制的妙用
    for (int i = 1; i <= n; i++)//二进制方式转换成01背包问题
        for(int j = m;j >= v[i];j--)
            f[j] = max(f[j],f[j-v[i]] + w[i]);
    cout << f[m] << endl;
    return 0;
}

多重背包III

有 N 种物品和一个容量为 V 的背包，每种物品最多 s 件，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N $\le$ 1000

0 < V $\le$ 20000

0 < $v_i,w_i,s_i$ $\le$ 20000

多重背包的单调队列优化：

#include<bits/stdc++.h>

using namespace std;

const int N = 2e4 + 10;
int n,m;
int f[N],g[N],q[N];
//f[i][j] = max(f[i - 1][j - v] + w,......),其中优化成一维后,
int main()//我们需要是上一次的[j-v],而我们当前层的[j-v]已经被更新过了(去掉一维的情况下)
{//那么我们就需要再来一个数组(g)来存一下上一层的情况
    cin >> n >> m;
    for(int i = 0;i < n;i++)
    {
        int v,w,s;//体积，价值，数量
        cin >> v >> w >> s;
        memcpy(g,f,sizeof(f));
        for(int j = 0;j < v;j++)//枚举下余数
        {
            int hh = 0,tt = -1;
            for(int k = j;k <= m;k += v)
            {//队首是最大的数
                if(hh <= tt && q[hh] < k - s * v)   hh++;//判断是否超过了s个物品
                if(hh <= tt)    f[k] = max(f[k],g[q[hh]] + (k - q[hh]) / v * w);
                while(hh <= tt && g[q[tt]] - (q[tt] - j) / v * w <= g[k] - (k - j) / v * w) tt--;//(等差数列)动态计算
                q[++tt] = k;
            }
        }
    }

    cout << f[m];
    return 0;
}

混合背包

有 N 种物品和一个容量是 V 的背包，问将哪些物品放入背包可使价值最大？

物品一共有三类：

  • 第一类物品只能用1次（01背包）
  • 第二类物品可以用无限次（完全背包）
  • 第三类物品最多只能用 s 次（多重背包）

数据范围：

0 < N,V $\le$ 1000
0 < vi,wi $\le$ 1000
−1 $\le$ si $\le$ 1000

#include<bits/stdc++.h>

using namespace std;

const int N = 1010;
int n,m;
int f[N];

int main()
{
    cin >> n >> m;

    for(int i = 0;i < n;i++)
    {
        int v,w,s;
        cin >> v >> w >> s;
        if(!s)//完全背包
        {
            for(int j = v;j <= m;j++)   
                f[j] = max(f[j],f[j - v] + w);
        }
        else
        {
            if(s == -1) s = 1;//说明当前为0,1背包
            for(int k = 1;k <= s;k *= 2)
            {
                for(int j = m;j >= k * v;j--)
                {
                    f[j] = max(f[j],f[j - k * v] + k * w);
                }
                s -= k;
            }
            if(s)
            {
                for(int j = m;j >= s * v;j--)
                    f[j] = max(f[j],f[j - s * v] + s * w);
            }
        }
    }
    cout << f[m] << endl;
    return 0;
}

分组背包

有 N 组物品和一个容量是 V 的背包，每组物品有若干个，同一组内的物品最多只能选一个。问将哪些物品放入背包可使价值最大？

数据范围

0 < N,V $\le$ 100
0 < Si $\le$ 100
0 < vij,wij $\le$ 100

思路：

细节：需要先循环体积，再循环决策。

体积优化成一维之后，要保证代码和优化之前是等价的，交换顺序之后就不等价了。

假设对第i组里的物品k，把所有体积遍历了一次，然后选择了这件物品，这个时候f[i] [j]已经包含了第i组中一个物品，也就是k；接下来还是这一组，第k+1个物品，又枚举所有体积，并且你也选择了这一个物品，即

f[i] [j]=max(f[i-1] [j],f[i-1] [j-v[i] [k+1]]+w[i] [k+1])中后者更大，更新之后f变成了后者
这就意味着在第i组中同时选择了k和k+1两个物品
而如果固定体积，遍历物品，就不会有这个问题，因为每次遍历都只能选中一件物品；
这本质上还是在优化成一维数组之后数组的更新过程中需要注意的问题，就像y总说的要保证等价。

#include<bits/stdc++.h>
using namespace std;
const int N = 110;
int n,m;
int v[N][N],w[N][N],s[N];
int f[N];
int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> s[i];
        for(int j = 0;j < s[i];j++)
            cin >> v[i][j] >> w[i][j];
    }
    for(int i = 1; i <= n;i++)
        for(int j = m;j >= 0;j--)//需要从上一状态转移，故从大到小，不太好确定最小值就用了大于等于0，毕竟下面有个if支撑
            for(int k = 0;k < s[i];k++)
                    if(v[i][k] <= j)    f[j] = max(f[j],f[j - v[i][k]] + w[i][k]);
    cout << f[m] << endl;
    return 0;
}

二维费用的背包问题

有 N 件物品和一个容量是 V 的背包，背包能承受的最大重量是 M。每件物品只能用一次。体积是 vi，重量是 mi，价值是 wi。求解将哪些物品装入背包，可使物品总体积不超过背包容量，总重量不超过背包可承受的最大重量，且价值总和最大。

数据范围：

0 < N $\le$ 1000
0 < V,M $\le$ 100
0 < vi,mi $\le$ 100
0 < wi $\le$ 1000

#include<bits/stdc++.h>

using namespace std;

const int N = 110;
int n,V,M;
int f[N][N];

int main()
{
    cin >> n >> V >> M;
    for(int i = 0;i < n;i++)
    {
        int v,m,w;
        cin >> v >> m >> w;
        for(int j = V;j >= v;j--)
        {
            for(int k = M;k >= m;k--)
            {
                f[j][k] = max(f[j][k],f[j - v][k - m] + w);
            }
        }
    }
    cout << f[V][M];
    return 0;
}

有依赖的背包问题

题目描述：

有 N 个物品和一个容量是 V 的背包。

物品之间具有依赖关系，且依赖关系组成一棵树的形状。如果选择一个物品，则必须选择它的父节点。

如下图所示：

如果选择物品5，则必须选择物品1和2。这是因为2是5的父节点，1是2的父节点。

每件物品的编号是 i，体积是 vi，价值是 wi，依赖的父节点编号是 pi。物品的下标范围是 1…N。

求解将哪些物品装入背包，可使物品总体积不超过背包容量，且总价值最大。

输出最大价值。

输入格式

第一行有两个整数 N，V，用空格隔开，分别表示物品个数和背包容量。

接下来有 N 行数据，每行数据表示一个物品。
第 i 行有三个整数 vi,wi,pi，用空格隔开，分别表示物品的体积、价值和依赖的物品编号。
如果 pi = −1，表示根节点。 数据保证所有物品构成一棵树。

输出格式

输出一个整数，表示最大价值。

数据范围

1 ≤ N,V ≤ 100
1 ≤ vi,wi ≤ 100

父节点编号范围：

  • 内部结点：1 ≤ pi ≤ N;
  • 根节点 pi=−1;

输入样例：

5 7
2 3 -1
2 2 1
3 5 1
4 7 2
3 6 2

输出样例：

11
#include<bits/stdc++.h>

using namespace std;

const int N = 110;
int n,m;
int v[N],w[N];
int h[N],e[N],ne[N],idx;
int f[N][N];//以i为根节点且体积不超过j的总价值是多少

void add(int a,int b)
{
    e[idx] = b,ne[idx] = h[a],h[a] = idx++;
}

void dfs(int u)
{
    for(int i = h[u];i != -1;i = ne[i])//循环物品组
    {
        int son = e[i];
        dfs(e[i]);

        //分组背包
        for(int j = m - v[u];j >= 0;j--)//循环体积(因为u是该树的根节点,所以必选,预留出u的空间)
            for(int k = 0;k <= j;k++)//循环决策
                f[u][j] = max(f[u][j],f[u][j - k] + f[son][k]);
    }

    //将物品u加进去
    for(int i = m;i >= v[u];i--)    f[u][i] = f[u][i - v[u]] + w[u];
    for(int i = 0;i < v[u];i++) f[u][i] = 0;//如果无法容纳物品u,那他的价值为0
}

int main()
{
    cin >> n >> m;

    memset(h,-1,sizeof(h));
    int root;
    for(int i = 1;i <= n;i++)
    {
        int p;
        cin >> v[i] >> w[i] >> p;
        if(p == -1) root = i;
        else    add(p,i);
    }

    dfs(root);

    cout << f[root][m] << endl;
    return 0;
}

背包问题求方案数

题目描述：

有 N 件物品和一个容量是 V 的背包。每件物品只能使用一次。

第 i 件物品的体积是 vi，价值是 wi。

求解将哪些物品装入背包，可使这些物品的总体积不超过背包容量，且总价值最大。

输出 最优选法的方案数。注意答案可能很大，请输出答案模 109 + 7 的结果。

输入格式

第一行两个整数，N，V，用空格隔开，分别表示物品数量和背包容积。

接下来有 N 行，每行两个整数 vi,wi，用空格隔开，分别表示第 ii 件物品的体积和价值。

输出格式

输出一个整数，表示 方案数 模 109+7 的结果。

数据范围

0 < N,V ≤ 1000
0 < vi,wi ≤ 1000

输入样例：

4 5
1 2
2 4
3 4
4 6

输出样例：

2
#include<bits/stdc++.h>

using namespace std;

const int N = 1010,mod = 1e9 + 7;
int n,m;
int f[N],g[N];

int main()
{
    cin >> n >> m;
    memset(f,-0x3f,sizeof(f));
    f[0] = 0;
    g[0] = 1;

    for(int i = 0;i < n;i++)
    {
        int v,w;
        cin >> v >> w;
        for(int j = m;j >= v;j--)
        {
            int maxv = max(f[j],f[j - v] + w);
            int cnt = 0;
            if(maxv == f[j])    cnt += g[j];
            if(maxv == f[j - v] + w)    cnt += g[j - v];
            g[j] = cnt % mod;
            f[j] = maxv;
        }
    }

    int res = 0;//因为定义的是体积恰好是j，所以需要循环一遍！
    for(int i = 0;i <= m;i++)   res = max(res,f[i]);
    int cnt = 0;

    for(int i = 0;i <= m;i++)
        if(res == f[i])
            cnt = (cnt + g[i]) % mod;

    cout << cnt << endl;

背包问题求具体方案

题目描述：

最长上升子序列II

题意：给定一个长度为 N（1 ≤ N ≤ 100000） 的数列，求数值严格单调递增的子序列的长度最长是多少。

二分：

案例：对于序列1 4 3，长度为2的上升子序列哪个更优呢？答案一定就是：1,3。因为如果后面有个更大的数（比如是4）的话它一定可以接到3的后面，而不能接到1,4的后面。

受此启发，我们可以处理出每个长度下，该序列结尾数最小的序列，显然该序列一定是单调递增的。当每增加一个数a，我们需要将它插入到一个小于a且最大的数的后面，二分即可。

#include<bits/stdc++.h>

using namespace std;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int n;
    cin >> n;

    vector<int> a(n + 1),q(n + 1);
    for(int i = 1;i <= n;i++)   cin >> a[i];

    int len = 0;
    for(int i = 1;i <= n;i++)
    {
        int l = 0,r = len;
        while (l < r)
        {
            int mid = l + r + 1 >> 1;
            if(q[mid] < a[i])   l = mid;
            else    r = mid - 1;
        }
        len = max(len,r + 1);
        q[r + 1] = a[i];
    }

    cout << len << "\n";
    return 0;
}

最长公共子序列

题意：给定两个长度分别为 N 和 M 的字符串 A 和 B，求既是 A 的子序列又是 B 的子序列的字符串长度最长是多少。

定义f[i] [j]：表示a的前i个字母，和b的前j个字母的最长公共子序列长度。

如果a[i] = b[j]，则：f[i] [j] = f[i - 1] [j - 1] + 1，则从f[i - 1] [j - 1]转移过来。

如果a[i] $\neq$ b[j]，则：f[i] [j]可以从f[i - 1] [j - 1]，f[i - 1] [j]，f[i] [j - 1]中转移过来，其中f[i - 1] [j - 1]已经被f[i] [j - 1]和f[i - 1] [j]包含了，故可以省略。

#include<bits/stdc++.h>

using namespace std;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    int n,m;
    cin >> n >> m;
    string a,b;
    cin >> a >> b;
    a = " " + a,b = " " + b;//令其下标从1开始

    vector f(n + 1,vector<int>(m + 1));
    for(int i = 1;i <= n;i++)
    {
        for(int j = 1;j <= m;j++)
        {
            if(a[i] == b[j])    f[i][j] = f[i - 1][j - 1] + 1;
            else    f[i][j] = max(f[i - 1][j],f[i][j - 1]);
        }
    }

    cout << f[n][m] << "\n";
    return 0;
}

最短编辑距离

给定两个字符串 A 和 B，现在要将 A 经过若干操作变为 B，可进行的操作有：

  1. 删除–将字符串 A 中的某个字符删除。
  2. 插入–在字符串 A 的某个位置插入某个字符。
  3. 替换–将字符串 A 中的某个字符替换为另一个字符。

现在请你求出，将 A 变为 B 至少需要进行多少次操作。

定义：f[i] [j]:将A的前i个字母变成B的前j个字母的最小操作次数。

1、删除:f[i] [j] = f[i - 1] [j] + 1。A的前i - 1个字母已经和B的前j个字母匹配了，故从此转移。

2、插入:f[i] [j] = f[i] [j - 1] + 1。A的前i个字母已经和B的前j - 1个字母匹配了，故从此转移。

3、替换:f[i] [j] = f[i - 1] [j - 1] + (a[i] != b[j])。A的前i - 1个字母已经和B的前j - 1个字母匹配了，最后看第i位和第j位是否相同。

#include<bits/stdc++.h>

using namespace std;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    int n,m;
    string a,b;
    cin >> n >> a >> m >> b;
    a = " " + a,b = " " + b;

    vector f(n + 1,vector<int>(m + 1));
    for(int i = 1;i <= n;i++)   f[i][0] = i;
    for(int i = 1;i <= m;i++)   f[0][i] = i;
    for(int i = 1;i <= n;i++)
        for(int j = 1;j <= m;j++)
        {
            f[i][j] = min(f[i - 1][j],f[i][j - 1]) + 1;
            f[i][j] = min(f[i][j],f[i - 1][j - 1] + (a[i] != b[j]));
        }

    cout << f[n][m] << "\n";
    return 0;
}

模板题：石子合并

题意： 有N堆石子，每次只能合并相邻的两堆石子，代价为这两堆石子的质量之和，问最终合并为一堆后的代价最小是多少。

区间合并：三重循环：

1、枚举区间长度

2、枚举左端点

3、枚举当前区间的中间点（由区间长度和左端点可知右端点，故枚举该区间的中间点）

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

constexpr ll INF = 1e18;

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    ll n;
    cin >> n;

    vector<int> a(n + 1);
    vector<ll> s(n + 1);
    for(int i = 1;i <= n;i++)   cin >> a[i],s[i] = s[i - 1] + a[i];
    //f[i][j]:将区间[i,j]合并的最小花费
    vector f(n + 1,vector<ll>(n + 1,INF));
    for(int i = 1;i <= n;i++)   f[i][i] = 0;
    for(int len = 1;len <= n;len++)
    {
        for(int l = 1;l + len - 1 <= n;l++)
        {
            int r = l + len - 1;
            for(int k = l;k < r;k++)//f[l,r] = f[l,k] + f[k + 1,r] + ...
            {
                f[l][r] = min(f[l][r],f[l][k] + f[k + 1][r] + s[r] - s[l - 1]);
            }
        }
    }

    cout << f[1][n] << "\n";
    return 0;
}

完全背包思想：价值为1,2,3,4,5,…n放到体积为n的背包中。

根据第i个物品选择的数量划分：

选择0个：f[i - 1] [j]

选择1个：f[i - 1] [j - i]

选择2个：f[i - 1] [j - 2i]

选择3个：f[i - 1] [j - 3i]

…

比如当前第i个物品选了2个:

即对应于:从1~i中选，i选了2个等价于f[i - 1] [j - 2i]，即数量相等

则:f[i] [j] = f[i - 1] [j] + f[i - 1] [j - i] + … + f[i - 1] [j - si]

用类似于完全背包的优化:f[i] [j - i] = f[i - 1] [j - i] + f[i - 1] [j - 2i] + …

故:f[i] [j] = f[i - 1] [j] + f[i] [j - i]

定义f[i] [j]:总和为i，由j个数组成的所有方案数。

1、最小值为1 -> f[i - 1] [j - 1]

2、最小值大于1 -> f[i - j] [j]

故f[i] [j] = f[i - 1] [j - 1] + f[i - j] [j]。

int dfs(int pos, int pre, int lead, int limit) {
    if (!pos) {
        边界条件
    }
    if (!limit && !lead && f[pos][pre] != -1) return f[pos][pre];
    int res = 0, up = limit ? a[pos] : 无限制位;
    for (int i = 0; i <= up; i ++) {
        if (不合法条件) continue;
        res += dfs(pos - 1, 未定参数, lead && !i, limit && i == up);
    }
    return limit ? res : (lead ? res : dp[pos][sum] = res);
}
int dp(int x) {
    memset(f, -1, sizeof f);
    int len = 0;

}

dp函数：

​ 每次初始化f数组为-1，长度len = 0。

​ a$_i$：每个数位具体数字。

dfs函数：

​ 变量res记录答案，初始化一般为0。

​ 变量up表示能枚举的最高位数。

​ 采用记忆化搜索的方式：if(!limit && !lead && f[pos] [pre] != -1) return f[pos] [pre];//只有无限制、无前导零才可以，不然都是未搜索完的情况。

必填参数：

int dfs(int pos,int num,int sum,int lead,int limit)
{
    if(!pos)
    {
        if(lead && !num)    return 1;
        return sum;
    }
    if(!lead && !limit && f[pos][sum] != -1)    return f[pos][sum];
    int res = 0,up = limit ? a[pos] : 9;

        {

            else    t = sum + 1;
        }
        else    t = sum;
        res += dfs(pos - 1,num,t,lead && i == 0,limit && i == up);
    }
    return limit ? res : (lead ? res : f[pos][sum] = res);
}

蒙德里安的梦想

题目链接：蒙德里安的梦想

核心：先把所有横着的摆放完后，竖着的就只有1种方案了（直接塞进去就行了）。

即：总方案数，等于只放横着的小方块的合法方案数。

那如何判断，当前方案是否合法呢？即看所有剩余位置，能否填充满竖着的小方块。又由于是竖着的小方块，所以可以按列来看，每一列内部所有连续的空着的小方块需要是偶数个！

so：

定义f[i] [j]：表示已经将前i - 1列摆好，且从第i - 1列伸出到第i列的状态为j的所有方案。

再考虑一些这个问题：假设第i - 2列伸出到第i - 1列的状态是k，第i - 1列伸出到第i列的状态为j，则什么时候k与j能拼在一起构成一个合法方案呢？

1、两个小方块不能处在同一行（如果处在同一行会有交集），即(j & k) == 0判断。

2、第i - 1列空着的小方格能被竖着的小方块塞满，即st[i | j]存在。

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

int n,m;

void solve()
{
    vector<bool> st((1 << n) + 1);
    //预处理所有合法的列
    for(int i = 0;i < 1 << n;i++)
    {
        bool ok = true;
        int cnt = 0;
        for(int j = 0;j < n;j++)
        {
            if(i >> j & 1)
            {
                if(cnt & 1)
                {
                    ok = false;
                    break;
                }
                cnt = 0;
            }
            else    cnt++;
        }
        if(cnt & 1) ok = false;
        st[i] = ok;
    }
    //预处理出所有合法的转移
    vector<int> state[(1 << n) + 1];
    for(int i = 0;i < 1 << n;i++)
        for(int j = 0;j < 1 << n;j++)
            if((i & j) == 0 && st[(i | j)])
                state[i].push_back(j);
	//状态转移计算
    vector f(m + 1,vector<ll>((1 << n) + 1));
    f[0][0] = 1;
    for(int i = 1;i <= m;i++)
        for(int j = 0;j < 1 << n;j++)
            for(auto k : state[j])
                f[i][j] += f[i - 1][k];
    
    cout << f[m][0] << "\n";
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    while(cin >> n >> m,n || m) solve();

    return 0;
}

最短Hamilton路径

f[i] [j]：状态为i的情况下，停到点j的最短路径是多少。

f[state] [j] = f[state_k] [k] + weight[k] [j]，state_k = state除掉j之后的集合，state_k要包含k。

#include<bits/stdc++.h>

using namespace std;

{

    cin >> n;
    for(int i = 0;i < n;i++)
        for(int j = 0;j < n;j++)  
            cin >> w[i][j];

    memset(f,0x3f,sizeof(f));
    f[1][0] = 0;
    for(int i = 0;i < 1 << n;i++)
        for(int j = 0;j < n;j++)
            if(i >> j & 1)
                for(int k = 0;k < n;k++)
                    if(i - (1 << j) >> k & 1)
                        f[i][j] = min(f[i][j],f[i - (1 << j)][k] + w[k][j]);

    cout << f[(1 << n) - 1][n - 1];
    return 0;
}

小国王

题目链接：小国王

题目描述

在 n×n 的棋盘上放 k 个国王，国王可攻击相邻的 8 个格子，求使它们无法互相攻击的方案总数。

输入格式

共一行，包含两个整数 n 和 k。

输出格式

共一行，表示方案总数，若不能够放置则输出0。

数据范围

1 ≤ n ≤ 10,
0 ≤ k ≤ n2

输入样例：

3 2

输出样例：

16
#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 12,M = 1 << 10,K = 110;
int n,m;
vector<int> state;//表示所有合法的状态
int cnt[M];//每个状态中1的个数
vector<int> head[M];//每个状态所有可以转移到的状态
ll f[N][K][M];
//集合：所有只摆在前i行，已经摆了j个国王，并且第i行摆放的状态是s的所有方案的集合
bool check(int state)
{
    for(int i = 0;i < n;i++)
        if((state >> i & 1) && (state >> i + 1 & 1))
            return false;
    return true;
}

int count(int state)//计算这个状态的二进制的表示中1的个数
{
    int res = 0;
    for(int i = 0;i < n;i++)    res += state >> i & 1;
    return res;
}

int main()
{
    cin >> n >> m;//n表示正方形的边长，m表示国王的数量

    for(int i = 0;i < 1 << n;i++)
        if(check(i))
        {
            state.push_back(i);
            cnt[i] = count(i);
        }
    
    for(int i = 0;i < state.size();i++)
        for(int j = 0;j < state.size();j++)
        {
            int a = state[i],b = state[j];//a表示第一个状态，b表示第二个状态
            if((a & b) == 0 && check(a | b))//(a & b) : 不在同一列上; (a | b) : 两行叠加起来不能有相邻的1
                head[i].push_back(j);//用的是下标
        }

    f[0][0][0] = 1;//初始化
    for(int i = 1;i <= n + 1; i++)//小技巧：枚举到n + 1
        for(int j = 0;j <= m;j++)
            for(int a = 0;a < state.size();a++)
                for(int b : head[a])
                {
                    int c = cnt[state[a]];
                    if(j >= c)
                        f[i][j][a] += f[i - 1][j - c][b];
                }
    cout << f[n + 1][m][0] << endl;//存在第n + 1行，但第n + 1行一个都没有摆
    return 0;
}

滑雪

#include<bits/stdc++.h>

using namespace std;

const int N = 310;

int n,m;
int f[N][N];
int g[N][N];

int dx[] = {-1,0,1,0},dy[] = {0,1,0,-1};

int dp(int x,int y)
{
    if(f[x][y] != -1)   return f[x][y];

    int& v = f[x][y];
    
    v = 1;
    for(int i = 0;i < 4;i++)
    {
        int nx = x + dx[i];
        int ny = y + dy[i];
        if(nx < 1 || nx > n || ny < 1 || ny > m || g[x][y] <= g[nx][ny])    continue;
        v = max(dp(nx,ny) + 1,v);
    }
    return v;
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    cin >> n >> m;
    for(int i = 1;i <= n;i++)
        for(int j = 1;j <= m;j++)
            cin >> g[i][j];

    memset(f,-1,sizeof(f));

    int res = 0;
    for(int i = 1;i <= n;i++)
        for(int j = 1;j <= m;j++)
            res = max(res,dp(i,j));

    cout << res << "\n";
    return 0;
}
