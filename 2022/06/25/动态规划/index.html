背包九讲

01背包

有 N 个物品和一个容量为 V 的背包，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N,V $\le$ 1000

0 < $v_i,w_i$ $\le$ 1000

二维数组（朴素版）：

状态f[i] [j] 定义为：前 i 个物品，背包容量 j 下的最优解（最大价值）。

有 N 件物品，则需要 N 次决策，每一次对第 i 件物品的决策，状态f[i] [j]不断由之前的状态更新而来。

1、当前背包容量不够（j < v[i]），没得选，因此前 i 个物品最优解即为前 i−1个物品最优解：

对应代码：f[i] [j] = f[i -1] [j]。

2、当前背包容量够，可以选，因此需要决策选与不选第 i 个物品：

选：f[i] [j] = f[i - 1] [j - v[i]] + w[i]。
不选：f[i] [j] = f[i - 1] [j] 。

我们的决策是如何取到最大价值，因此以上两种情况取 max() 。

#include<bits/stdc++.h>
using namespace std;
const int N = 1010;
int n,m;
int v[N],w[N];//v表示体积，w表示价值
int f[N][N];//f表示状态
int main()
{
    cin >> n >> m;//物品个数、背包容量
    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];//输入每个物品的体积、价值
    for(int i = 1;i <= n;i++)//f[0][0~m]表示考虑0件物品，总体积不超过0~m，故最大价值为0，全局变量以默认初始化，因此从1开始
        for(int j = 0;j <= m;j++)
        {
            f[i][j] = f[i-1][j];
            if(j >= v[i])   f[i][j] = max(f[i][j],f[i-1][j-v[i]] + w[i]);
        }
    cout << f[n][m] << endl;
    return 0;
}

一维数组（优化版）：

状态f[j]定义：N 件物品，背包容量j下的最优解。

注意枚举背包容量j必须从m开始。

为什么一维情况下枚举背包容量需要逆序？在二维情况下，状态f[i] [j]是由上一轮i - 1的状态得来的，f[i] [j]与f[i - 1] [j]是独立的。而优化到一维后，如果我们还是正序，则有f[较小体积]更新到f[较大体积]，则有可能本应该用第i-1轮的状态却用的是第i轮的状态（因为体积小的状态已经被更新了，当体积大的需要用到体积小的去更新时，用到的是第i轮，而非第i - 1轮）。

例如，一维状态第i轮对体积为 3 的物品进行决策，则f[7]由f[4]更新而来，这里的f[4]正确应该是f[i - 1] [4]，但从小到大枚举j这里的f[4]在第i轮计算却变成了f[i] [4]。当逆序枚举背包容量j时，我们求f[7]同样由f[4]更新，但由于是逆序，这里的f[4]还没有在第i轮计算，所以此时实际计算的f[4]仍然是f[i - 1] [4]。

故状态转移方程为：f[j] = max(f[j], f[j - v[i]] + w[i] 。

#include<bits/stdc++.h>

using namespace std;

const int N = 1010;
int n,m;
int v[N],w[N];//v表示体积，w表示价值
int f[N];//一维数组

int main()
{
    cin >> n >> m;//物品个数、背包容量
    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];//输入每个物品的体积、价值
    
    for(int i = 1;i <= n;i++)
        for(int j = m;j >= v[i];j--)//逆序更新
        {
            f[j] = max(f[j],f[j-v[i]] + w[i]);
        }
    cout << f[m] << endl;
    return 0;
}

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

void solve()
{
	int n;
	cin >> n;

	vector<ll> a(n + 1);
	for (int i = 1; i <= n; i++)	cin >> a[i];

	vector<int> pw(11);
	pw[0] = 1;
	for (int i = 1; i <= 10; i++)	pw[i] = pw[i - 1] * 3;

	vector<ll> dp(pw[10]);
	for (int i = 1; i <= n; i++)
	{
		auto f = dp;
		vector<int> cnt(11);
		auto t = a[i];
		while (t)
		{
			cnt[t % 10] ++;
			t /= 10;
		}
		
		function<void(int, int, int)> dfs = [&](int x, int last, int cur) -> void
		{
			if (x == 10)
			{
				f[cur] = max(f[cur], dp[last] + a[i]);
				return;
			}
			for (int i = 0; i + cnt[x] <= 2; i++)
				dfs(x + 1, last + i * pw[x], cur + (i + cnt[x]) * pw[x]);
		};

		dfs(0, 0, 0);
		swap(dp, f);
	}

	cout << dp.back() << "\n";
}

int main()
{
	cin.tie(nullptr)->sync_with_stdio(false);

	int t;
	cin >> t;

	while (t--)
	{
		solve();
	}

完全背包

有 n 种物品和一个容量为 v 的背包，每个物品有无限个，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N,V $\le$ 1000

0 < $v_i,w_i$ $\le$ 1000

二维数组（朴素版）：

与01背包的代码几乎完全一样，只不过枚举体积的顺序是从小到大开始！

公式推导过程：

f[i] [j] = max(f[i - 1] [j],f[i - 1] [j - v] + w,f[i - 1] [j - 2v] + 2w,f[i - 1] [j - 3v] + 3w…)

然后突然有了一个“神奇”的表达式，f[i] [j - v]如下：

f[i] [j - v] = max(f[i - 1] [j - v],f[i - 1] [j - 2v] + w,f[i - 1] [j - 3v] + 2w,…)

不难看出，标亮的式子中：上面比下面多加了一个w。

因此：f[i] [j] = max(f[i - 1] [j],f[i] [j - v] + w)。

#include<iostream>

using namespace std;

const int N = 1010;
int n,m;
int f[N][N];
int v[N],w[N];

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m;

    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i];

    for(int i = 1;i <= n;i++)
    {
        for(int j = 0;j <= m;j++)
        {
            f[i][j] = f[i - 1][j];
            if(j >= v[i])   f[i][j] = max(f[i][j],f[i][j - v[i]] + w[i]);
        }
    }

    cout << f[n][m] << "\n";
    return 0;
}

f[i] [j] = max(f[i - 1] [j],f[i] [j - v] + w) 完全背包

可以发现，完全背包需要用到当前的状态，即从小到大枚举。

多重背包

多重背包I

有 N 种物品和一个容量为 V 的背包，每种物品最多 s 件，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N,V $\le$ 100

0 < $v_i,w_i,s_i$ $\le$ 100

朴素版：

#include<bits/stdc++.h>
using namespace std;
const int N = 110;
int n,m;
int v[N],w[N],s[N];
int f[N][N];
int main()
{
    cin >> n >> m;
    for(int i = 1;i <= n;i++)   cin >> v[i] >> w[i] >> s[i];
    for(int i = 1;i <= n;i++)
        for(int j = 0;j <= m;j++)
            for(int k = 0;k <= s[i] && k * v[i] <= j;k++)
                f[i][j] = max(f[i][j],f[i-1][j-v[i]*k] + w[i] * k);
    cout << f[n][m] << endl;
    return 0;
}

多重背包II

有 N 种物品和一个容量为 V 的背包，每种物品最多 s 件，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N $\le$ 1000

0 < V $\le$ 2000

0 < $v_i,w_i,s_i$ $\le$ 2000

多重背包的二进制优化：

利用二进制优化，时间复杂度可以从 O(N M S) 降到O(N logS M ),从4∗10^9^降到了2∗10^7^。

思想：将每种物品拆分成1,2,4,8,16,32…，即分成1,2,4，…，2^k^,s这些个物品，其中s小于2^k+1^。这样可以利用01背包的思想，拼凑出任意数量的解（包括最优解），然后调用01背包模板即可。

举例：如果某种物品数量s = 200，则可拆分成：1,2,4,8,16,32,64,73（73 < 128）。

#include<bits/stdc++.h>
using namespace std;
const int N = 12010,M = 2010;//一共有1000个物品，每个物品最多拆分成log2000个物品，故是N的由来
int n,m;
int v[N],w[N];
int f[M];
int main()
{
    cin >> n >> m;
    int cnt = 0;
    for (int i = 1; i <= n; i++)
    {
        int a,b,s;
        cin >> a >> b >> s;
        int k = 1;
        while (k <= s)
        {
            cnt++;
            v[cnt] = a * k;
            w[cnt] = b * k;
            s -= k;
            k *= 2;
        }
        if(s > 0)
        {
            cnt++;
            v[cnt] = a * s;
            w[cnt] = b * s;
        }
    }
    n = cnt;//每个数选与不选都能拼凑出任意的数字，这就是2进制的妙用
    for (int i = 1; i <= n; i++)//二进制方式转换成01背包问题
        for(int j = m;j >= v[i];j--)
            f[j] = max(f[j],f[j-v[i]] + w[i]);
    cout << f[m] << endl;
    return 0;
}

多重背包III

有 N 种物品和一个容量为 V 的背包，每种物品最多 s 件，问将哪些物品放入背包可使价值最大？

数据范围：

0 < N $\le$ 1000

0 < V $\le$ 20000

0 < $v_i,w_i,s_i$ $\le$ 20000

多重背包的单调队列优化：

#include<bits/stdc++.h>

using namespace std;

const int N = 2e4 + 10;
int n,m;
int f[N],g[N],q[N];
//f[i][j] = max(f[i - 1][j - v] + w,......),其中优化成一维后,
int main()//我们需要是上一次的[j-v],而我们当前层的[j-v]已经被更新过了(去掉一维的情况下)
{//那么我们就需要再来一个数组(g)来存一下上一层的情况
    cin >> n >> m;
    for(int i = 0;i < n;i++)
    {
        int v,w,s;//体积，价值，数量
        cin >> v >> w >> s;
        memcpy(g,f,sizeof(f));
        for(int j = 0;j < v;j++)//枚举下余数
        {
            int hh = 0,tt = -1;
            for(int k = j;k <= m;k += v)
            {//队首是最大的数
                if(hh <= tt && q[hh] < k - s * v)   hh++;//判断是否超过了s个物品
                if(hh <= tt)    f[k] = max(f[k],g[q[hh]] + (k - q[hh]) / v * w);
                while(hh <= tt && g[q[tt]] - (q[tt] - j) / v * w <= g[k] - (k - j) / v * w) tt--;//(等差数列)动态计算
                q[++tt] = k;
            }
        }
    }

    cout << f[m];
    return 0;
}

混合背包

有 N 种物品和一个容量是 V 的背包，问将哪些物品放入背包可使价值最大？

物品一共有三类：

  • 第一类物品只能用1次（01背包）
  • 第二类物品可以用无限次（完全背包）
  • 第三类物品最多只能用 s 次（多重背包）

数据范围：

0 < N,V $\le$ 1000
0 < vi,wi $\le$ 1000
−1 $\le$ si $\le$ 1000

#include<bits/stdc++.h>

using namespace std;

const int N = 1010;
int n,m;
int f[N];

int main()
{
    cin >> n >> m;

    for(int i = 0;i < n;i++)
    {
        int v,w,s;
        cin >> v >> w >> s;
        if(!s)//完全背包
        {
            for(int j = v;j <= m;j++)   
                f[j] = max(f[j],f[j - v] + w);
        }
        else
        {
            if(s == -1) s = 1;//说明当前为0,1背包
            for(int k = 1;k <= s;k *= 2)
            {
                for(int j = m;j >= k * v;j--)
                {
                    f[j] = max(f[j],f[j - k * v] + k * w);
                }
                s -= k;
            }
            if(s)
            {
                for(int j = m;j >= s * v;j--)
                    f[j] = max(f[j],f[j - s * v] + s * w);
            }
        }
    }
    cout << f[m] << endl;
    return 0;
}

分组背包

有 N 组物品和一个容量是 V 的背包，每组物品有若干个，同一组内的物品最多只能选一个。问将哪些物品放入背包可使价值最大？

数据范围

0 < N,V $\le$ 100
0 < Si $\le$ 100
0 < vij,wij $\le$ 100

思路：

细节：需要先循环体积，再循环决策。

体积优化成一维之后，要保证代码和优化之前是等价的，交换顺序之后就不等价了。

假设对第i组里的物品k，把所有体积遍历了一次，然后选择了这件物品，这个时候f[i] [j]已经包含了第i组中一个物品，也就是k；接下来还是这一组，第k+1个物品，又枚举所有体积，并且你也选择了这一个物品，即

f[i] [j]=max(f[i-1] [j],f[i-1] [j-v[i] [k+1]]+w[i] [k+1])中后者更大，更新之后f变成了后者
这就意味着在第i组中同时选择了k和k+1两个物品
而如果固定体积，遍历物品，就不会有这个问题，因为每次遍历都只能选中一件物品；
这本质上还是在优化成一维数组之后数组的更新过程中需要注意的问题，就像y总说的要保证等价。

#include<bits/stdc++.h>
using namespace std;
const int N = 110;
int n,m;
int v[N][N],w[N][N],s[N];
int f[N];
int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> s[i];
        for(int j = 0;j < s[i];j++)
            cin >> v[i][j] >> w[i][j];
    }
    for(int i = 1; i <= n;i++)
        for(int j = m;j >= 0;j--)//需要从上一状态转移，故从大到小，不太好确定最小值就用了大于等于0，毕竟下面有个if支撑
            for(int k = 0;k < s[i];k++)
                    if(v[i][k] <= j)    f[j] = max(f[j],f[j - v[i][k]] + w[i][k]);
    cout << f[m] << endl;
    return 0;
}

二维费用的背包问题

有 N 件物品和一个容量是 V 的背包，背包能承受的最大重量是 M。每件物品只能用一次。体积是 vi，重量是 mi，价值是 wi。求解将哪些物品装入背包，可使物品总体积不超过背包容量，总重量不超过背包可承受的最大重量，且价值总和最大。

数据范围：

0 < N $\le$ 1000
0 < V,M $\le$ 100
0 < vi,mi $\le$ 100
0 < wi $\le$ 1000

#include<bits/stdc++.h>

using namespace std;

const int N = 110;

{

    for(int i = 0;i < n;i++)
    {

有依赖的背包问题

#include<bits/stdc++.h>

using namespace std;

const int N = 110;
int n,m;
int v[N],w[N];
int h[N],e[N],ne[N],idx;
int f[N][N];//以i为根节点且体积不超过j的总价值是多少

void add(int a,int b)
{
    e[idx] = b,ne[idx] = h[a],h[a] = idx++;
}

void dfs(int u)
{
    for(int i = h[u];i != -1;i = ne[i])//循环物品组
    {
        int son = e[i];
        dfs(e[i]);

        //分组背包
        for(int j = m - v[u];j >= 0;j--)//循环体积(因为u是该树的根节点,所以必选,预留出u的空间)

{

    memset(f,-0x3f,sizeof(f));
    f[0] = 0;
    g[0] = 1;

    for(int i = 0;i < n;i++)
    {
        int v,w;
        cin >> v >> w;
        for(int j = m;j >= v;j--)
        {
            int maxv = max(f[j],f[j - v] + w);
            int cnt = 0;
            if(maxv == f[j])    cnt += g[j];
            if(maxv == f[j - v] + w)    cnt += g[j - v];

            f[j] = maxv;

    for(int i = 0;i <= m;i++)   res = max(res,f[i]);
    int cnt = 0;

    for(int i = 0;i <= m;i++)
        if(res == f[i])
            cnt = (cnt + g[i]) % mod;

    cout << cnt << endl;
    return 0;
}

  • 插入：$f[i][j] = f[i][j - 1] + 1$

定义$f[i][j]$：总和为 $i$ ，由 $j$ 个数组成的所有方案数。

  • 定义 $dp[i][j][k]$：前 $i$ 个数，用了 $j$ 个数，且总和为 $k$ 的方案数。
  • 集合划分：第 $i$ 个数选 / 不选：

    2. 不选：$dp[i][j][k] = dp[i - 1][j][k]$

    for (int i = 0; i <= n; i++) // 初始化，前 i 个数，选了 0 个数，总和为 0 的方案数为 1。

                    if (k >= i)
                    {

                    }
                }
            }

换根DP

题意：给定一个 $n$ 个点的树，请求出一个结点，使得以这个结点为根时，所有结点的深度之和最大。

思路：两遍 $dfs$ 。

第一遍 $dfs$ 用来预处理，处理每个子树的大小和每个点的深度；

第二遍 $dfs$ 考虑状态转移，即体现 “换根“ 的地方。令 $f_u$ 为以 $u$ 为根时，所有结点的深度之和。

$f_v \leftarrow f_u$ 可以体现为换根，即以 $u$ 为根转移到以 $v$ 为根。显然在换根的转移过程中，以 $v$ 为根或以 $u$ 为根会导致其子树中的结点的深度产生改变。具体表现为：

  • 所有在 $u$ 的子树上的结点深度都减少了 $1$，那么总深度和就减少了 $s_v$；
  • 所有不在 $v$ 的子树上的结点深度都增加了 $1$，那么总深度和就增加了 $n - s_v$；

因此，可以推出状态转移方程：$f_v = f_u - s_v + n - s_v = f_u + n - 2 × s_v$ 。

模板题代码：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

蒙德里安的梦想

题目链接：蒙德里安的梦想

核心：先把所有横着的摆放完后，竖着的就只有1种方案了（直接塞进去就行了）。

即：总方案数，等于只放横着的小方块的合法方案数。

那如何判断，当前方案是否合法呢？即看所有剩余位置，能否填充满竖着的小方块。又由于是竖着的小方块，所以可以按列来看，每一列内部所有连续的空着的小方块需要是偶数个！

so：

定义f[i] [j]：表示已经将前i - 1列摆好，且从第i - 1列伸出到第i列的状态为j的所有方案。

再考虑一些这个问题：假设第i - 2列伸出到第i - 1列的状态是k，第i - 1列伸出到第i列的状态为j，则什么时候k与j能拼在一起构成一个合法方案呢？

1、两个小方块不能处在同一行（如果处在同一行会有交集），即(j & k) == 0判断。

2、第i - 1列空着的小方格能被竖着的小方块塞满，即st[i | j]存在。

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

int n,m;

void solve()
{
    vector<bool> st((1 << n) + 1);
    //预处理所有合法的列
    for(int i = 0;i < 1 << n;i++)
    {
        bool ok = true;
        int cnt = 0;
        for(int j = 0;j < n;j++)
        {
            if(i >> j & 1)
            {
                if(cnt & 1)
                {
                    ok = false;
                    break;
                }
                cnt = 0;
            }
            else    cnt++;
        }
        if(cnt & 1) ok = false;
        st[i] = ok;
    }
    //预处理出所有合法的转移
    vector<int> state[(1 << n) + 1];
    for(int i = 0;i < 1 << n;i++)
        for(int j = 0;j < 1 << n;j++)
            if((i & j) == 0 && st[(i | j)])
                state[i].push_back(j);
	//状态转移计算

    for(int i = 1;i <= m;i++)
        for(int j = 0;j < 1 << n;j++)
            for(auto k : state[j])
                f[i][j] += f[i - 1][k];
    
    cout << f[m][0] << "\n";

    int l = 1, r = 0;
    vector<int> cnt(n + 1);
    auto modify = [&](int L, int R) -> void
    {
        while (r < R)   res += cnt[a[++r]]++;
        while (l > L)   res += cnt[a[--l]]++;
        while (r > R)   res -= --cnt[a[r--]];
        while (l < L)   res -= --cnt[a[l++]];
    };
    function<void(int, int, int, int ,int)> solve = [&](int k, int l, int r, int ql, int qr) -> void
    {
        if (l > r)  return ;
        int mid = l + r >> 1, qm;
        ll mn = 1e18;

        for (int i = ql; i <= qr; i++)
        {
            modify(i, mid);
            if (f[k - 1][i - 1] + res < mn)
            {
                mn = f[k - 1][i - 1] + res;
                qm = i;
            }
        }
        f[k][mid] = mn;
        solve(k, l, mid - 1, ql, qm);
        solve(k, mid + 1, r, qm, qr);
    };

状态定义：f[i] [j]：以 $a[j]$ 为结尾的长度为 $i$ 的上升子序列的个数

状态转移：$f[i][j]= \sum_{k<j,a[k]<a[j]}f[i-1][k]$

具体来说：在外层循环 $i$，建立一个树状数组，以 $a[k]$ 为下标存储 $f[i - 1][k]$ 的值。当内层循环到 $j$ 时，$f[i][j] += ask(a[j] - 1)$，然后在转移到下一个 $j$ 之前 $add(a[j], f[i - 1][j])$ 。$j$ 从小到大循环保证了 $k < j$，查询 $f[i - 1][j - 1]$ 的前缀和保证了 $a[k] < a[j]$。

时间复杂度：$O(NMlogN)$

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

constexpr int N = 3e4 + 10;

int n;
int a[N], tr[N];

int lowbit(int x)
{
    return x & -x;
}

void add(int x, int v)
{
    for (int i = x; i <= n; i += lowbit(i))
        tr[i] += v;
}

int sum(int x)
{
    int res = 0;
    for (int i = x; i; i -= lowbit(i))
        res += tr[i];
    return res;
}

不太好想但又可以想到的DP定义：C. Road Optimization

定义：$f[i][j]$：前 $i$ 个标志，第 $i$ 个标志未被移除，且已经移除了 $j$ 个标志的答案集合。

转移：枚举前一个位置 $k$ ：$f[i][j] = f[k][j - (i - 1 - k)] + (i - 1 - k) * v[k]$，其中 $(i - 1 - k)$ 是从 $k$ 到 $i$ 位置内，所有的移除个数。

时间复杂度：$O(n^3)$

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

void solve()
{
    int n, m, k;
    cin >> n >> m >> k;

    vector<int> d(n + 2);
    for (int i = 1; i <= n; i++)    cin >> d[i];
    d[n + 1] = m;

    vector<ll> v(n + 1);
    for (int i = 1; i <= n; i++)    cin >> v[i];

    vector<vector<ll>> f(n + 2, vector<ll>(n + 1, 1e18));
    f[1][0] = 0;
    for (int i = 2; i <= n + 1; i++)
    {
        for (int j = 0; j <= k; j++)
        {
            for (int z = 1;z <= i - 1; z++)
            {
                int t = i - 1 - z;
                ll w = v[z] * (d[i] - d[z]);
                if (t > j)  continue;
                f[i][j] = min(f[i][j], f[z][j - t] + w);
            }
        }
    }

    ll res = 1e18;
    for (int i = 0; i <= k; i++)
        res = min(res, f[n + 1][i]);

    cout << res << "\n";
}

    return 0;

    for (int i = 1; i <= 2023; i++)
    {
        s += to_string(i);
    }

    vector<ll> f(5); // 2, 20, 202, 2023
    
    for (int i = 0; i < s.size(); i++)
    {
        if (s[i] == '2')    f[0] ++, f[2] += f[1];
        else if (s[i] == '0')   f[1] += f[0];
        else if (s[i] == '3') f[3] += f[2];
    }

    cout << f[3] << "\n";
    return 0;
}
