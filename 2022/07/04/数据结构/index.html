三分

三分时间复杂度：O（n$^2$）

三分与二分的思想类似，也是通过不断缩小答案可能的区间范围，来找到最终答案所在的位置。

三分一般被用来查找某类单峰函数的峰值。

三分需要找三等分点：M1 = (R - L) / 3 + L,M2 = (R - L) / 3 + 2L。

注意：

但也有某类单峰函数不能用三分！

某些问题的函数可能是单峰的，但峰值两边并非严格递增（递减）的，即有可能出现”平台“（）一条与x轴平行的直线。

什么时候使用三分？

​ 即一个函数，在考虑的范围[l,r]内是单峰的，并且从峰值往左右两边都是要严格单调递增或严格单调递减的。

整数三分

E. Restorer Distance

while(l + 2 < r)需要区间可以分成三段！

最后暴力枚举l ~ r之间的结果即可！

数三分求凹函数：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 1e5 + 10;

int n,A,R,M;
int a[N];

ll f(int x)
{
    ll y = 0,z = 0;
    for(int i = 1;i <= n;i++)
    {
        if(a[i] < x)    y += x - a[i];
        else    z += a[i] - x;
    }

    ll res = min(y,z) * M; 
    if(y > z)   res += (y - z) * A;
    else    res += (z - y) * R;
    return res;
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);

    cin >> n >> A >> R >> M;
    M = min(M,A + R);
    for(int i = 1;i <= n;i++)   cin >> a[i];

    int l = 0,r = 1e9;
    while (l + 2 < r)
    {
        int m1 = (r - l) / 3 + l;
        int m2 = (r - l) / 3 * 2 + l;
        if(f(m1) > f(m2))   l = m1;
        else    r = m2;
    }
    
    ll res = f(l);
    for(int i = l + 1;i <= r;i++)   res = min(res,f(i));

    cout << res << "\n";
    return 0;
}

浮点数三分

【模板】三分法

模板记忆口诀：哪个情况更糟就删去哪一部分（每次会删去三分之一的区间）：

比如要求最大值，就要把小的那一部分给删去；

要求最小值，则把大的那一部分删去。

#include<bits/stdc++.h>

using namespace std;

constexpr double eps = 1e-7;

int main()
{

    {
        double res = 0;
        for(int i = 1;i <= n;i++)
            res += a[i] * pow(x,n - i + 1);

        double m1 = (r - l) / 3 + l;
        double m2 = (r - l) / 3 * 2 + l;

        else    l = m1;
    }

    return 0;
}

KMP

KMP算法是可以在文本串s中快速查找模式串p的一种算法。

封装模板：

auto KMP = [&](string p,string s) -> vector<int>
{
    vector<int> res;
    int n = p.size(),m = s.size();
    p = " " + p,s = " " + s;

    vector<int> ne(n + 1);
    for(int i = 2,j = 0;i <= n;i++)
    {
        while(j && p[i]!=p[j+1])    j = ne[j];
        if(p[i]==p[j+1])    j++;
        ne[i] = j;
    }

    for(int i = 1,j = 0;i <= m;i++)
    {
        while(j && s[i]!=p[j+1])    j = ne[j];
        if(s[i]==p[j+1])    j++;
        if(j==n)//匹配成功
        {
            res.push_back(i - n);//匹配的下标从0开始
            j = ne[j];
        }
    }
    return res;
};

typedef long long ll;

const int N = 2e7 + 7;
int n, m, z[N], p[N];
char a[N], b[N];

void Z(char *s, int n) 
{
	z[1] = n;

	for (int i = 2, l = 0, r = 0; i <= n; i++) 
    {
		if (i <= r) z[i] = min(z[i - l + 1], r - i + 1);
		while (i + z[i] <= n && s[i + z[i]] == s[z[i] + 1]) ++z[i];
		if (i + z[i] - 1 > r) l = i, r = i + z[i] - 1;
	}
}

		if (i + p[i] - 1 > r) l = i, r = i + p[i] - 1;

Manacher算法

应用：求序列的最长回文子串的长度是多少？

算法需知：

  • 在原串中，不管是奇数还是偶数的回文串，在新串中都对应长度为奇数的回文串。
  • 细节方面：若原串空间为 $n(即数组下标为[0,n - 1])$，则新串空间长度需为 $2n + 3(即数组下标为[0,2n+2])$。
  • 具体而言，只需考虑 $[1,n]$，其中 $n$ 为更新后的长度，即 $n{新} = 2n{原} + 1$，为奇数，该区间串的内容为：”#a#a#”(以 $aa$ 字符串为例)。

数组含义：

  • a[i]：原串
  • s[i]：新串
  • p[i]：新串中以 $i$ 为中心的最大回文串的半径是多少。

时间复杂度：$O(n)$

模板：

const int N = 1e7 + 10;

int n;
char a[N << 1],s[N << 1];
int p[N << 1];//以i为中心的最大回文半径

void init()
{
    n = 0;
    int len = strlen(a);
    s[0] = '$';
    for(int i = 0;i < len;i++) s[++n] = '#',s[++n] = a[i];
    s[++n] = '#',s[n + 1] = '^';
}

void manacher()
{
    init();
    int mid = 0,r = 0;
    for(int i = 1;i <= n;i++)
    {
        p[i] = i <= r ? min(p[mid * 2 - i],r - i + 1) : 1;
        while(s[i - p[i]] == s[i + p[i]])   p[i]++;
        if(i + p[i] - 1 > r)    r = i + p[i] - 1,mid = i;
    }
}

template<typename T>
struct DSU {
    vector<T> p, sz;
    DSU(T n) : p(n), sz(n, 1) { iota(p.begin(), p.end(), 0); }
    void init(int n) {//初始化
    p.resize(n);
    iota(p.begin(), p.end(), 0);
    sz.assign(n, 1);
    }
    T find(T x) { //找父节点
       while (x != p[x]) x = p[x] = p[p[x]];
       return x;
    }
    bool merge(T x, T y) { //合并集合
       x = find(x);
       y = find(y);
       if (x == y) return false;
       sz[x] += sz[y];
       p[y] = x;
       return true;
    }
    bool same(T x, T y) { return find(x) == find(y); }//判断是否在同一集合
    T size(T x) { return sz[find(x)]; }//获取集合大小
};

启发式合并

基本思想：将小的集合，合并到大的集合里。

时间复杂度：$O(nlogn)$

模板题讲解：

  • 题意：
    • 给点 $n$ 个数，定义数字相同的为同一颜色，连续相同颜色的数为一段。询问1：将所有数 $x$ 修改为 $y$；询问2：一共有多少段？
  • 思路：
    1. 如何统计答案？
      • 定义两个临界，即 $a[0]$ 与 $a[n]$ 颜色都为 $0$，则临界位置是和上一个位置的颜色是不同的（通俗讲就是第二段颜色的第一个和第一段颜色的最后一个的颜色一定是不相同的），同一段颜色内部（不算第一个）是和上一个位置颜色是相同的。故基于此想法，假设应该有 $n$ 段，用公式：$ans += a[i] \neq a[i - 1]$ 后，可以得到 $n - 1$ 段，为了减少处理边界的情况，人为定义 $a[0]$ 与 $a[n]$ 段是一个不存在的颜色，即又增加了 $2$ 段，故最终正确段数答案应为 $ans - 1$。
    2. 将颜色 $x$ 合并到颜色 $y$ 中。
      • 若 $size|x| \le size|y|$，正常处理即可；
      • 若 $size|x| > size|y|$，则会将 $y$ 合并到 $x$ 中，而中间有个映射转换，下面将细讲：
        • 将 $y$ 中的颜色对应下标都放到 $x$ 中，并将对应下标都改成与 $x$ 一致，此时原数组中的所有本是 $y$ 颜色的却全部变成了 $x$ 的颜色，这里暂时没什么问题；
        • 下次如果再用到 $y$ 颜色，本身应该是有 $y$ 颜色的，但原数组是没有 $y$ 颜色的， 但是 $pos[y]$ 是有大小的（上一次是将 $pos[x]$ 给清空了，很符合 将 $x$ 合并成 $y$），因此映射转换在这里。

{
    u.sum = l.sum + r.sum;
    u.lmax = max(l.lmax,l.sum + r.lmax);
    u.rmax = max(r.rmax,r.sum + l.rmax);
    u.tmax = max(max(l.tmax,r.tmax),l.rmax + r.lmax);
}

{

}

{

    else

}

    {

​ 所以每个数最多开5次根号即可变成1，因此我们需要维护一个区间最大值，当最大值为1时，之后就不再进行修改了。所以可以在modify递归的时候去”暴力“进行计算取根号，递归到叶子节点后修改下数值，一但一段区间的最大值为1，就不用再递归修改了。

typedef long long ll;

const int N = 2e5 + 10;

struct Node
{
    int l,r;
    ll mx,mi;//区间最大、最小值
    ll sum,lazy;//懒标记:区间加
}tr[N << 2];

void pushup(int u)
{
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
    tr[u].mx = max(tr[u << 1].mx,tr[u << 1 | 1].mx);
    tr[u].mi = min(tr[u << 1].mi,tr[u << 1 | 1].mi);
}

void eval(int u,ll add)
{
    tr[u].lazy += add;
    tr[u].mi += add,tr[u].mx += add;
    tr[u].sum += (tr[u].r - tr[u].l + 1) * add;
}

void pushdown(int u)
{
    if(tr[u].lazy)
    {
        eval(u << 1,tr[u].lazy);
        eval(u << 1 | 1,tr[u].lazy);
        tr[u].lazy = 0;
    }
}

{

}

}

RMQ（Range Maximum/Minimum Query）问题，即区间最值查询问题，是指对于长度为 n 的数组a，查询数组a中下标在 [ i , j ] 里的最大值或最小值。

ST算法是一个非常有名的在线处理RMQ问题的算法，它在 O ( n log ⁡ n ) 时间内进行预处理，然后可以在 O ( 1 ) 时间内回答每个查询，其本质是倍增与动态规划的思想。要注意的是：ST算法只适用于静态区间求最值。

int LG2[N];

void init()
{
    LG2[1] = 0;
    for(int i = 2;i <= n;i++)   LG2[i]  =LG2[i >> 1] + 1;

    for(int j = 0;j <= LG2[n];j++)//类似区间DP，循环长度
        for(int i = 1;i + (1 << j) - 1 <= n;i++)//因为集合含义是从i开始,长度为2^j的区间的最大值,右边界是i + 2^j - 1
            if(!j)  f[i][j] = a[i];
            else    f[i][j] = max(f[i][j - 1],f[i + (1 << j - 1)][j - 1]);  
    return ;
}//f[i][j] = max(f[i][j - 1],f[i + 2^(j - 1)][j - 1]);

int query(int l,int r)
{
    int k = LG2[r - l + 1];
    return max(f[l][k],f[r - (1 << k) + 1][k]);
}

struct ODT//珂朵莉树
{
    struct Node
    {
        int l,r;
        mutable T v;
        Node(int _l,int _r = -1,T _v = 0):l(_l),r(_r),v(_v){}
        bool operator<(const Node&o) const {return l < o.l;}
    };

    void insert(int l,int r,T v){//往set中插入结点
        S.insert(Node(l,r,v));
    }
    auto split(int pos){//断开区间
        auto it = S.lower_bound(Node(pos));
        if(it != S.end() && it->l == pos)   return it;
        --it;
        int l = it->l,r = it->r;
        T v = it->v;
        S.erase(it);
        S.insert(Node(l,pos - 1,v));
        return S.insert(Node(pos,r,v)).first;
    }
    void add(int l,int r,T v){  //区间加
        for(auto end = split(r + 1),it = split(l);it != end;it++)
            it->v += v;
    }
    void assign(int l,int r,T v){   //区间赋值
        auto end = split(r + 1),begin = split(l);
        S.erase(begin,end);
        S.insert(Node(l,r,v));
    }
    T get_k(int l,int r,int k){ //区间第k大的数
        vector<pair<T,int>> v;
        for(auto end = split(r + 1),it = split(l);it != end;it++){
            v.push_back({it->v,it->r - it->l + 1});
        }
        sort(v.begin(),v.end());
        for(auto x : v){
            k -= x.second;
            if(k <= 0)  return x.first;
        }
        return -1;
    }
    T sum_pow(int l,int r,int k,int p){ //区间[l,r]的k次幂之和
        ll res = 0;
        for(auto end = split(r + 1),it = split(l);it != end;it++){
            res = (res + 1ll * (it->r - it->l + 1) * fpow(it->v,k,p) % p) % p;
        }
        return res;
    }
};

#include<bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10,INF = 1e8;

int n;
struct Node
{
    int l,r;//左右边界
    int key,val;//key:二叉搜索树的权值;val:堆的权值
    int cnt,size;//这个数出现多少次;size:每个子树里面一共有多少个数,方便求排名
}tr[N];

int root,idx;//root:表示根节点;idx:当前分配到了第几个节点

void pushup(int p)//用p的两个儿子来算p的信息
{
    tr[p].size = tr[tr[p].l].size + tr[tr[p].r].size + tr[p].cnt;
}

int get_node(int key)//创建并取得节点
{
    tr[++idx].key = key;
    tr[idx].val = rand();//随机值
    tr[idx].cnt = tr[idx].size = 1;//创建的时候默认是叶节点
    return idx;
}

void build()//初始化
{
    get_node(-INF),get_node(INF);//插入两个哨兵
    root = 1,tr[1].r = 2;//根节点是1号点,1号点右儿子是2号点
    pushup(root);//已经有两个值了,pushup一下
}

void zig(int &p)    //右旋
{
    int q = tr[p].l;//q是p的儿子
    tr[p].l = tr[q].r,tr[q].r = p,p = q;//p的左儿子变成q的右儿子
    pushup(tr[p].r),pushup(p);
}

void zag(int &p)    //左旋
{
    int q = tr[p].r;
    tr[p].r = tr[q].l,tr[q].l = p,p = q;
    pushup(tr[p].l),pushup(p);
}

void insert(int &p,int key)//记得传引用
{
    if(!p)  p = get_node(key);
    else if(tr[p].key == key)   tr[p].cnt++;
    else if(tr[p].key > key)    
    {
        insert(tr[p].l,key);
        if(tr[tr[p].l].val > tr[p].val) zig(p);//左子树的值大于当前的值,右旋一下 
    }
    else//同理
    {
        insert(tr[p].r,key);
        if(tr[tr[p].r].val > tr[p].val) zag(p);
    }
    pushup(p);//更新一下当前的值
}

void remove(int &p,int key)
{
    if(!p)  return ;//如果要删除的值不存在,就不用管
    if(tr[p].key == key)
    {
        if(tr[p].cnt > 1)   tr[p].cnt --;
        else if(tr[p].l || tr[p].r)//如果不是叶子节点的话
        {
            if(!tr[p].r || tr[tr[p].l].val > tr[tr[p].r].val)//如果右子树是空的或者左子树的val大于右子树的val
            {
                zig(p);//右旋
                remove(tr[p].r,key);//删除右子树
            }
            else
            {
                zag(p);
                remove(tr[p].l,key);
            }
        }
        else    p = 0;//是叶子节点,直接删除
    }
    else if(tr[p].key > key)    remove(tr[p].l,key);//去左边删
    else    remove(tr[p].r,key);//去右边删

    pushup(p);//更新p
}

树链剖分

算法流程：
1、通过dfs1确定深度、父节点、子树大小；

​ 2、通过dfs2确定dfs序：id[]、dfs序所对应的权值：nw[]、重链的顶点：top[]；

​ 3、树链剖分的思想类似于最近公共祖先（lca），其是通过重链往上爬的，确定的dfs序：id[]方便得知线段树对应的区间，从而将树上路径的问题转换成线段树区间上的问题。

树链剖分的作用：

把树转化成一个序列，这样树中 任意一条路径 就转换到 log⁡n 段的 连续区间。

然后就可以使用各种 维护区间信息 的 数据结构了。

细节：

dfs序：优先遍历重儿子。

树链剖分中的几个定义：

重/轻儿子：节点个数最多的子树，该节点为其父节点的 重儿子，其余都为 轻儿子（叶节点没有轻重儿子概念）。

重/轻边：重儿子和它父节点的边称为 重边，轻儿子和它父节点的边称为 轻边。

重链：由 重边 构成的 极大路径。

模板题代码：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 1e5 + 10,M = 1e6 + 10;

int n,m;
int w[N],h[N],e[M],ne[M],idx;
int id[N],nw[N],cnt;//id[]:dfs序编号  nw[]:每个编号的点的权值(新编号);fa[]:每个点的父节点;son[]每个点的重儿子是谁
int dep[N],sz[N],top[N],fa[N],son[N];//dep[]:每个点的深度;sz:以每个点为根的子树的大小,top[]:每个点所在重链的顶点;
struct Tree                          
{
    int l,r;
    ll add,sum;//add:懒标记;sum:区间和
}tr[N << 2];

void add(int a,int b)
{
    e[idx] = b,ne[idx] = h[a],h[a] = idx++;
}

void dfs1(int u,int father,int depth)
{
    dep[u] = depth,fa[u] = father,sz[u] = 1;//初始化一些数据:深度、父节点、子树大小
    for(int i = h[u] ;~i;i = ne[i])//枚举这个点的所有儿子
    {
        int j = e[i];
        if(j == father) continue;//不要往上搜
        dfs1(j,u,depth + 1);
        sz[u] += sz[j];
        if(sz[son[u]] < sz[j])  son[u] = j;//重儿子是子树节点最多的儿子
    }
}//确定dep[],sz[],fa[],son[]数组

void dfs2(int u,int t)//t:当前点所在重链的顶点
{//初始化: 给它一个dfs序;  dfs序中第cnt个点的权值是w[u];  当前点所在重链顶点是t;
    id[u] = ++cnt,nw[cnt] = w[u],top[u] = t;
    if(!son[u]) return;//当前节点是叶节点(没有重儿子),直接return
    dfs2(son[u],t);//优先搜索重儿子
    for(int i = h[u]; ~i;i = ne[i])//接下来是轻儿子
    {
        int j = e[i];
        if(j == fa[u] || j == son[u])   continue;//不能往上搜，不能是重儿子
        dfs2(j,j);//轻儿子所在重链顶点就是他自己
    }
}//确定id[],nw[],top[]

void pushup(int u)
{
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
}

void eval(Tree &u,ll add)
{
    u.add += add;
    u.sum += (u.r - u.l + 1) * add;
}

void pushdown(int u)
{
    eval(tr[u << 1],tr[u].add);
    eval(tr[u << 1 | 1],tr[u].add);
    tr[u].add = 0;
}

void build(int u,int l,int r)
{
    tr[u] = {l,r,0,nw[r]};
    if(l == r)  return;
    int mid = l + r >> 1;
    build(u << 1,l,mid),build(u << 1 | 1,mid + 1,r);
    pushup(u);
}

void update(int u,int l,int r,int k)
{
    if(l <= tr[u].l && r >= tr[u].r)    eval(tr[u],k);
    else
    {
        pushdown(u);
        int mid = tr[u].l + tr[u].r >> 1;
        if(l <= mid)    update(u << 1,l,r,k);
        if(r > mid) update(u << 1 | 1,l,r,k);
        pushup(u);
    }
}

ll query(int u,int l,int r)
{
    if(l <= tr[u].l && r >= tr[u].r)    return  tr[u].sum;
    pushdown(u);
    int mid = tr[u].l + tr[u].r >> 1;
    ll res = 0;
    if(l <= mid)    res += query(u << 1,l,r);
    if(r > mid) res += query(u << 1 | 1,l,r);
    return res;
}

void update_path(int u,int v,int k)//"爬山法"
{
    while (top[u] != top[v])//两个点不在同一条重链里
    {
        if(dep[top[u]] < dep[top[v]])   swap(u,v);//交换完之后u所在重链的顶点会靠低一些
        update(1,id[top[u]],id[u],k);//更新需要跳的区间
        u = fa[top[u]];//走完之后u要跳到重链的上面
    }
    if(dep[u] < dep[v]) swap(u,v);//u在v的下面
    update(1,id[v],id[u],k);//最后更新一下区间
}

ll query_path(int u,int v)
{
    ll res = 0;
    while (top[u] != top[v])//两个点不在同一条重链里
    {
        if(dep[top[u]] < dep[top[v]])   swap(u,v);
        res += query(1,id[top[u]],id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u,v);
    res += query(1,id[v],id[u]);
    return res;
}

void update_tree(int u,int k)
{
    update(1,id[u],id[u] + sz[u] - 1,k);//id[u]是根节点编号,先是重儿子,之后是轻儿子,故右区间是id[u] + sz[u] - 1
}

ll query_tree(int u)
{
    return query(1,id[u],id[u] + sz[u] - 1);
}

int main()
{
    scanf("%d",&n);
    for(int i = 1;i <= n;i++)   scanf("%d",&w[i]);
    memset(h,-1,sizeof(h));
    for(int i = 0;i < n - 1;i++)
    {
        int a,b;
        scanf("%d%d",&a,&b);
        add(a,b),add(b,a);
    }
    dfs1(1,-1,1);//求每个点的重儿子是谁;第一个参数是点的编号;第二个参数是:该点的父节点;第三个参数是:每个点的深度是多少
    dfs2(1,1);//求dfs序
    build(1,1,n);//建立一颗线段树

    scanf("%d",&m);
    while (m--)
    {
        int t,u,v,k;
        scanf("%d%d",&t,&u);
        if(t == 1)//将路径u ~ v之间所有节点的权值加k
        {
            scanf("%d%d",&v,&k);
            update_path(u,v,k);
        }
        else if(t == 2)//将以u为根的子树上所有节点的权值加k
        {
            scanf("%d",&k);
            update_tree(u,k);
        }
        else if(t == 3)//询问节点u ~ v之间路径所有节点的权值和
        {
            scanf("%d",&v);
            printf("%lld\n",query_path(u,v));
        }
        else    printf("%lld\n",query_tree(u));//询问以节点 u 为根的子树上的所有节点的权值和
    }
    
    return 0;
}

动态树问题：

维护一个森林，支持删除某条边，加入某条边，并保证加边，删边后仍是森林。我们要维护该森林的一些信息。
一般的操作有两点连通性， 两点路径权值和， 连接两点 和 切断某条边、修改信息 等

LCT 则是用多个 Splay 来维护 多个实链，Splay 的特性就使得我们可以进行 树的合并、分割 操作。

零散知识点：

将所有边分为虚边和实边，用splay维护所有实边路径，其中序遍历就是要维护的路径。

splay本质上：维护所有实边，用splay中的后继与前驱来维护原树中的父子关系。

虚边：是用splay的根节点来维护。

实链剖分：

把树拆分成若干条实链，用splay这种动态数据结构来维护实链。通过对实链的拆分与组合，构造答案。

一个节点只能选一个儿子做实儿子，其他都是虚儿子。

实边：父节点与实儿子之间的边，是双向边。

虚边：由虚儿子指向父节点的边，是单向边。（认父不认子）

实链：由实边构成的链。每条实链的节点深度是严格递增的。

辅助树：由若干splay构成，与原树是等价的，只需要维护辅助树。

性质：每条实链用一个splay维护，中序遍历按深度严格递增。

注意：区分树根和splay的根。

一些辅助树构成了Link Cut Tree(LCT)，其维护的是整个森林。

【模板】动态树（Link Cut Tree）

给定 n 个点以及每个点的权值，要你处理接下来的 m个操作。
操作有四种，操作从 0 到 3 编号。点从 1 到 n 编号。

  • 0 x y 代表询问从 x 到 y 的路径上的点的权值的 xor 和。保证 x 到 y 是联通的。
  • 1 x y 代表连接 x 到 y，若 x 到 y 已经联通则无需连接。
  • 2 x y 代表删除边 (x,y)，不保证边 (x,y) 存在。
  • 3 x y 代表将点 x上的权值变成 y。

代码模板：

#include<iostream>
#include<cstring>
#include<algorithm>

using namespace std;

const int N = 1e5 + 10;

int n,m;
struct Node
{
    int s[2],p,v;
    int sum,rev;//sum:异或和,rev:旋转标记;
    //这里的懒标记表示:当前结点已经旋转过了，但我的子树还未旋转(即我这个点的状态是懒标记操作之后的状态)
}tr[N];
int stk[N];

void pushrev(int x)
{
    swap(tr[x].s[0],tr[x].s[1]);
    tr[x].rev ^= 1;
}

void pushup(int x)
{
    tr[x].sum = tr[tr[x].s[0]].sum ^ tr[x].v ^ tr[tr[x].s[1]].sum;
}

void pushdown(int x)
{
    if(tr[x].rev)
    {
        pushrev(tr[x].s[0]),pushrev(tr[x].s[1]);
        tr[x].rev = 0;
    }
}

bool isroot(int x)  //判断当前点是不是根节点
{
    return tr[tr[x].p].s[0] != x && tr[tr[x].p].s[1] != x;
}

void rotate(int x)
{
    int y = tr[x].p,z = tr[y].p;
    int k = tr[y].s[1] == x;
    if(!isroot(y))  tr[z].s[tr[z].s[1] == y] = x;//如果y是根节点则不能更新y和它父节点的关系
    tr[x].p = z;
    tr[y].s[k] = tr[x].s[k ^ 1],tr[tr[x].s[k ^ 1]].p = y;
    tr[x].s[k ^ 1] = y,tr[y].p = x;
    pushup(y),pushup(x);
}

void splay(int x)   //将x转到根节点
{//可能有从下往上的操作(即不是从上往下递归而来的),所以需要先将路径存到栈里，然后从上往下pushdown一遍
    int top = 0,r = x;
    stk[++top] = r;
    while(!isroot(r))   stk[++top] = r = tr[r].p;
    while(top)  pushdown(stk[top--]);
    while(!isroot(x))
    {
        int y = tr[x].p,z = tr[y].p;
        if(!isroot(y))
            if((tr[y].s[1] == x) ^ (tr[z].s[1] == y))   rotate(x);
            else    rotate(y);
        rotate(x);
    }
}

void access(int x)  //建立一条从根到x的(实边)路径，同时将x变成splay的根节点
{
    int z = x;
    for(int y = 0;x;y = x,x = tr[x].p)//x沿着实边往上走
    {
        splay(x);
        tr[x].s[1] = y,pushup(x);
    }
    splay(z);
}

void makeroot(int x)    //将x变成原树的根节点
{
    access(x);
    pushrev(x);
}

int findroot(int x) //找到x所在原树的根节点，再将原树的根节点旋转到splay的根节点
{
    access(x);
    while(tr[x].s[0])   pushdown(x),x = tr[x].s[0];
    splay(x);
    return x;
}

void split(int x,int y) //给x和y之间的路径建立一个splay，其根节点是y
{
    makeroot(x);
    access(y);
}

void link(int x,int y)  //如果x和y不连通，则加入一条x和y之间的边
{
    makeroot(x);
    if(findroot(y) != x)    tr[x].p = y;
}

void cut(int x,int y)   //如果x和y之间存在边，则删除该边
{
    makeroot(x);
    if(findroot(y) == x && tr[y].p == x && !tr[y].s[0])
    {
        tr[x].s[1] = tr[y].p = 0;
        pushup(x);
    }
}

int main()
{
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> n >> m;

    for(int i = 1;i <= n;i++)   cin >> tr[i].v;
    while (m--)
    {
        int t,x,y;
        cin >> t >> x >> y;
        if(t == 0)
        {
            split(x,y);
            cout << tr[y].sum << "\n";
        }
        else if(t == 1) link(x,y);
        else if(t == 2) cut(x,y);
        else
        {
            splay(x);
            tr[x].v = y;
            pushup(x);
        }
    }
    
    return 0;
}

【模板】AC 自动机（简单版）

代码：

#include <bits/stdc++.h>

using namespace std;

const int N = 1e4 + 10, S = 55, M = 1e6 + 10;

int n;//tr[][26]:第一维要看所有字符串总和的最大值是多少
int tr[N * S][26], cnt[N * S], idx; // tr[N*S][26]:结点个数是N*S，最多是26个字母;cnt[N*S]:以每个结点结尾的单词的数量
char str[M];
int q[N * S], ne[N * S];

void insert()
{
    int p = 0; //从根节点开始，0是根
    for (int i = 0; str[i]; i++)
    {
        int t = str[i] - 'a'; //偏移量
        if (!tr[p][t])  tr[p][t] = ++idx; //如果不存在，就创建
        p = tr[p][t];         //现在这条路一定就有了(一定存在)，直接走过去
    }
    cnt[p]++; //以p结尾的单词数量加1
}

void build() //宽搜
{
    int hh = 0, tt = -1;
    for (int i = 0; i < 26; i++)
        if (tr[0][i])   q[++tt] = tr[0][i];
    while (hh <= tt) //经典宽搜
    {
        int t = q[hh++];
        for (int i = 0; i < 26; i++)
        {
            int &p = tr[t][i];
            if(!p)  p = tr[ne[t]][i];
            else
            {
                ne[p] = tr[ne[t]][i];//我们就让它的fail(ne)指针指向他父亲节点 t 的 fail(ne) 指针指向的那个节点
                q[++tt] = p;
            }
        }
    }
}

int main()
{
    int T;
    scanf("%d", &T);
    while (T--) // T组测试数据
    {           //开始需先清空
        memset(tr, 0, sizeof(tr));
        memset(cnt, 0, sizeof(cnt));
        memset(ne, 0, sizeof(ne));
        idx = 0;
        scanf("%d", &n); // n个单词
        for (int i = 0; i < n; i++)
        {
            scanf("%s", str);
            insert();
        }
        build();
        scanf("%s", str);
        int res = 0; //定义答案，表示匹配单词的数量
        for (int i = 0, j = 0; str[i]; i++)
        {
            int t = str[i] - 'a';
            j = tr[j][t];

            int p = j;
            while (p && ~cnt[p])
            {
                res += cnt[p];
                cnt[p] = -1;
                p = ne[p];
            }
        }
        printf("%d\n", res);
    }
    return 0;
}

后缀自动机

功能非常强大的数据结构，主要应用如下：

  1. 求模式串 $p$ 有没有在字符串 $s$ 中出现过。
  2. 求模式串 $p$ 在字符串 $s$ 中出现的次数。
  3. 求字符串 $s$ 的最小表示法（循环同构字符串中字典序最小的一个）。
  4. 求 $s$ 的本质不同子串个数。
  5. 最长公共子串。

模板：

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int N = 2e6 + 10;

int tot = 1,last = 1;//tot:当前用到哪个点了;last:上一个点的编号;1号点是一个空点
struct Node
{
    int len,fa;//这个状态里所有子串的最大长度;fa:绿色边(link)
    int ch[26];//所有儿子
}node[N];
char str[N];
ll f[N],ans;
int h[N],e[N],ne[N],idx;

void extend(int c)
{
    int p = last,np = last = ++tot;//p为上一次插入的状态,np表示我们当前插入的新状态
    f[tot] = 1;
    node[np].len = node[p].len + 1;
    for(;p && !node[p].ch[c];p = node[p].fa)//存在父子关系节点包含的子串有后缀关系
        node[p].ch[c] = np;//只要p不是空的并且p是没有c这个儿子的,那么p就往前走
    if(!p)  node[np].fa = 1;//整个到根节点的路径全都没有c这个儿子,特判一下,将其置为1(空,即连向起始点)
    else
    {
        int q = node[p].ch[c];//用q表示当前这个点的c这条边
        if(node[q].len == node[p].len + 1)   node[np].fa = q;
        else
        {//看它听不听话,如果不听话就换一个人！
            int nq = ++ tot;//新建一个新的q
            node[nq] = node[q],node[nq].len = node[p].len + 1;//新的q完全把旧的q复制过来
            node[q].fa = node[np].fa = nq;//把np指向新的听话的这个人
            for(;p && node[p].ch[c] == q;p = node[p].fa)    node[p].ch[c] = nq;
        }//把所有p的c这个儿子替换成听话的这个
    }
}

void add(int a,int b)
{
    e[idx] = b,ne[idx] = h[a],h[a] = idx++;
}

void dfs(int u)
{
    for(int i = h[u]; ~i;i = ne[i])
    {
        dfs(e[i]);
        f[u] += f[e[i]];
    }
    if(f[u] > 1)    ans = max(ans,f[u] * node[u].len);
}

int main()
{
    scanf("%s",str);
    for(int i = 0;str[i];i++)   extend(str[i] - 'a');//将新的字符扩展一遍
    memset(h,-1,sizeof(h));
    for(int i = 2;i <= tot;i++) add(node[i].fa,i);//1号点是没有父节点的;当前点的父节点向当前点连一条边
    dfs(1);
    printf("%lld\n",ans);
}

支配树

定义：

对于一张有向图，我们确定一个起点 $S$，对于一个点 $k$，称 $x$ 为其支配点当且仅当，删去点 $x$ 后，$S$ 无法到达 $k$，即 $S$ 到达 $k$ 的所有路径都必须通过 $x$。

$Lengauer-Tarjan$ 算法分为三步：

  1. 求出 $dfs$ 树，得到每个点 $k$ 的 $dfs$ 序 $d_k$。
  2. 求解半支配点。
  3. 求解支配点。

时间复杂度：$O(nlogn)$

使用指南：

$idom[i]$ 数组：$i$ 的支配点是谁，即 $i$ 被谁支配；

$vector> domain_tree$：支配树（DAG），支配树上的点会被从根节点到该点上路径上所有的点所支配。

封装模板：

struct DominationTree//支配树
{
   int n,idx;
   vector<vector<int>> g,fg,hg,domain_tree;//原图,反图,半支配树,支配树
   vector<int> dfn,ord,fa;//ord:dfs序列;
   vector<int> uni,mn,idom,sdom;//idom为支配点

   DominationTree(int _n):n(_n),idx(0),g(n + 1),fg(n + 1),hg(n + 1),domain_tree(n + 1)
   {
       dfn.resize(n + 1),ord.resize(n + 1),fa.resize(n + 1);
       uni.resize(n + 1),mn.resize(n + 1),idom.resize(n + 1),sdom.resize(n + 1);
   }
   void add(int a,int b)
   {
       g[a].push_back(b);
       fg[b].push_back(a);
   }
   void tarjan(int u)
   {
       ord[dfn[u] = ++idx] = u;
       for(auto v : g[u])
           if(!dfn[v])
           {
               fa[v] = u;
               tarjan(v);
           }
   }
   int query_uni(int u)
   {
       if(u == uni[u]) return u;
       int res = query_uni(uni[u]);
       if(dfn[sdom[mn[uni[u]]]] < dfn[sdom[mn[u]]])    mn[u] = mn[uni[u]];
       return uni[u] = res;
   }
   //求以st为起点的所有点的支配点
   void contract(int st)
   {
       tarjan(st);
       for(int i = 1;i <= n;i++)   sdom[i] = uni[i] = mn[i] = i;
       for(int i = idx;i >= 2;i--)
       {
           int u = ord[i];
           for(auto v : fg[u])
           {
               if(!dfn[v]) continue;
               query_uni(v);
               if(dfn[sdom[mn[v]]] < dfn[sdom[u]]) sdom[u] = sdom[mn[v]];
           }
           uni[u] = fa[u];//半支配点的求解过程
           hg[sdom[u]].push_back(u);//维护半支配树
           for(auto v : hg[u = fa[u]])
           {
               query_uni(v);
               idom[v] = u == sdom[mn[v]] ? u : mn[v];
           }
           hg[u].clear();
       }

       for(int i = 2;i <= idx;i++)
       {
           int u = ord[i];
           if(idom[u] ^ sdom[u])   idom[u] = idom[idom[u]];
       }

       //建立支配树
       for(int i = idx;i >= 2;i--)
           domain_tree[idom[ord[i]]].push_back(ord[i]);
       }
};

int main()
{
    DominationTree dtree(n);
    dtree.contract(1);//调用(1为起点)
}
